278  public static Scalar Parse(string str)
279  {
280  Scalar s = new Scalar(0d);
281  int pos = Read(ref s,str);
282  if (pos < 0)
283  throw new FormatException("String does not represent a scalar. Culture must be neutral.");
284  else
285  {
286  pos = Parser.SkipSpaces(str,pos);
287  if (pos >= 0 && pos < str.Length)
288  throw new FormatException("Non-space characters follow the scalar.");
289  }
290  return s;
291  }
