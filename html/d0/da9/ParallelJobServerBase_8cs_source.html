503  protected void Serve()
504  {
505  try
506  {
507  if (_outputLevel >= 1)
508  {
509  Console.WriteLine("Server " + Id + ": server thread started, sleep time = " + SleepTimeMs + "ms");
510  }
511  _isServerRunning = true;
512  bool doServe = true;
513  while (doServe)
514  {
515  bool runJob = false;
516  lock (Lock)
517  {
518  runJob = DoRunJobByServer;
519  }
520  if (runJob)
521  {
522  if (_outputLevel >= 1)
523  {
524  if (_jobData != null)
525  {
526  Console.WriteLine(" Server " + Id + ": job requested, running job " + _jobData.Id + ", dispatcher ID = "
527  + _jobData.DispatcherJobId);
528  } else
529  {
530  Console.WriteLine(" Server " + Id + ": job requested, null container.");
531  }
532  }
533  DoRunJobByServer = false;
534  RunJobInThread();
535  }
536  if (CommandStopServing)
537  {
538  doServe = false;
539  CommandStopServing = false;
540  }
541  Thread.Sleep(SleepTimeMs);
542  }
543  }
544  catch
545  {
546  // Remark: Thread abortion is already handled in TunJobInThread.
547  if (_outputLevel >= 1)
548  {
549  Console.WriteLine("Server " + Id + ": serving thread aborted (exception thrown).");
550  }
551  }
552  finally
553  {
554  lock (Lock)
555  {
556  _isServerRunning = false;
557  _workingThread = null;
558  }
