378  public static void GenerateCoordinatesPlain(StructuredField3d<vec3> field,
379  double minxRef, double maxxRef,
380  double minyRef, double maxyRef,
381  double minzRef, double maxzRef)
382  {
383  double dimx = field.Dim1, dimy = field.Dim2, dimz = field.Dim3;
384  double hx = 0, hy = 0, hz = 0, xRef, yRef, zRef;
385  if (dimx > 1)
386  hx = (maxxRef - minxRef) / (double)(dimx - 1);
387  if (dimy > 1)
388  hy = (maxyRef - minyRef) / (double)(dimy - 1);
389  if (dimz > 1)
390  hz = (maxzRef - minzRef) / (double)(dimz - 1);
391  for (int k = 0; k < dimz; ++k)
392  for (int j = 0; j < dimy; ++j)
393  for (int i = 0; i < dimx; ++i)
394  {
395  xRef = minxRef + i * hx;
396  yRef = minyRef + j * hy;
397  zRef = minzRef + k * hz;
398  field[i, j, k] = new vec3(xRef, yRef, zRef);
399  }
400  }
