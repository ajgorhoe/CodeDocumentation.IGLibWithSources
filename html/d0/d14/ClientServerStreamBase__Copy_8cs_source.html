337  public void GetRequestOrResponse(ref string responseOrRequestString, out bool isMessage,
338  out string messageOrCommandName, out string [] messageArguments)
339  {
340  isMessage = false;
341  messageOrCommandName = null;
342  messageArguments = null;
343  if (responseOrRequestString != null)
344  {
345  if (responseOrRequestString.StartsWith(MessagePrefix))
346  {
347  if (responseOrRequestString.StartsWith(MessagePrefixWithSeparator))
348  {
349  // String is actually a message:
350  isMessage = true;
351  responseOrRequestString = responseOrRequestString.Substring(MessagePrefixWithSeparator.Length);
352  string[] parts = responseOrRequestString.Split(new char[] {' ', '\t', '\r', '\n'}, StringSplitOptions.RemoveEmptyEntries);
353  messageOrCommandName = parts[0];
354  int numParts = parts.Length;
355  if (numParts > 1)
356  {
357  messageArguments = new string[numParts - 1];
358  for (int i = 1; i < numParts; ++i)
359  messageArguments[i-1] = parts[i];
360  }
361  } else
362  {
363  // String begins with message prfix, but it is not a message (because a separator is not following):
364  // Actual request/response string is obtained by excluding the inserted additional character after the MessagePrefix:
365  responseOrRequestString = MessagePrefix + responseOrRequestString.Substring(MessagePrefix.Length + 1);
366  }
367  }
368  }
