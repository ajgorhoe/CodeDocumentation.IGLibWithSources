334  public static color Average(bool clipComponents, params color[] colors)
335  {
336  double r = 0, g = 0, b = 0, opacity = 0;
337  int length = 0;
338  if (colors != null)
339  length = colors.Length;
340  for (int i = 0; i < length; ++i)
341  {
342  color current = colors[i];
343  r += current.R;
344  g += current.G;
345  b += current.B;
346  opacity += current.Opacity;
347  }
348  r /= (double)length;
349  g /= (double)length;
350  b /= (double)length;
351  if (clipComponents)
352  {
353  if (r < 0) r = 0;
354  if (r > 1) r = 1;
355  if (g < 0) g = 0;
356  if (g > 1) g = 1;
357  if (b < 0) b = 0;
358  if (b > 1) b = 1;
359  if (opacity < 0) opacity = 0;
360  if (opacity > 1) opacity = 1;
361  }
362  return new color(r, g, b, opacity);

437  public static color Mixture(bool clipComponents, double[] weights, color[] colors)
438  {
439  double r = 0, g = 0, b = 0, opacity = 0;
440  int length = 0;
441  if (colors != null)
442  {
443  length = colors.Length;
444  if (weights == null)
445  throw new ArgumentException("Weights for color mixing are not specified (null reference), although colors are.");
446  if (weights.Length != length)
447  throw new ArgumentException("Number of weights (" + weights.Length + ") is different than number of colors ("

547  public static color Scale(color[] colors, double value)
548  {
549  if (colors==null)
550  throw new ArgumentNullException("colors", "Colors not specified (null reference).");
551  int num = colors.Length;
552  if (num == 0)
553  throw new ArgumentException("There are no colors specified (empty array).");
554  else if (num == 1)
555  return colors[0];
556  double h = 1.0 / ((double)num - 1.0); // Length of interval between two marker colors
557  double hFactor = value / h;
558  int cellIndex = (int) Math.Floor(hFactor);
559  if (cellIndex < 0)
560  return colors[0];
561  else if (cellIndex >= (num - 1))
562  return colors[num - 1];
563  else
564  {
565  double cellValue = hFactor - (double) cellIndex; // running from 0 to 1 within a cell
566  color col1 = colors[cellIndex];
567  color col2 = colors[cellIndex + 1];
568  if (cellValue < 0) cellValue = 0;
569  if (cellValue > 1) cellValue = 1;
570  double red = 0, green = 0, blue = 0, opacity = 0;
571  red = (1.0 - cellValue) * col1.R + cellValue * col2.R;
572  green = (1.0 - cellValue) * col1.G + cellValue * col2.G;
573  blue = (1.0 - cellValue) * col1.B + cellValue * col2.B;
574  opacity = (1.0 - cellValue) * col1.Opacity + cellValue * col2.Opacity;
575  if (red < 0) red = 0;
576  if (red > 1) red = 1;
577  if (green < 0) green = 0;
578  if (green > 1) green = 1;
579  if (blue < 0) blue = 0;
580  if (blue > 1) blue = 1;
581  if (opacity < 0) opacity = 0;
582  if (opacity > 1) opacity = 1;
583  return new color(red, green, blue, opacity);
584  }
