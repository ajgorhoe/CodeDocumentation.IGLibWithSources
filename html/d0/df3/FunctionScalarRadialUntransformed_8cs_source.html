188  public override void HessianPlain(IVector x, IMatrix hessian)
189  {
190  int dim = x.Length;
191  double normx = x.Norm;
192  double fDer = Function.Derivative(normx);
193  double fSecondDer = Function.SecondDerivative(normx);
194  if (normx < Epsilon)
195  {
196  // Close to zero, hessian matrix is diagonal, diagonal elements equal to second derivatives
197  // of 1-D function from which this scalar function is derived.
198  hessian.SetZero();
199  for (int i = 0; i < dim; ++i)
200  hessian[i, i] = fSecondDer;
201  } else
202  {
203  double normxSquare = normx * normx;
204  double normxCube = normxSquare * normx;
205  double element;
206  // Nondiagonal terms:
207  for (int i=1; i<dim; ++i)
208  for (int j = 0; j < i; ++j)
209  {
210  double prod_x_i_j = x[i]*x[j];
211  element = fSecondDer * prod_x_i_j / normxSquare
212  - fDer * prod_x_i_j / normxCube;
213  hessian[i,j] = element;
214  hessian[j, i] = element;
215  }
216  for (int i = 0; i < dim; ++i)
217  {
218  double x_i_square = x[i]; // square of component
219  x_i_square *= x_i_square;
220  hessian[i,i]=fSecondDer*x_i_square/normxSquare
221  + fDer*((1/normx)+(x_i_square/normxCube));
222  }
223  }
224  }
