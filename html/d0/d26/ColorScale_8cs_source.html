334  public override color GetContinuousReferenceColor(double referenceValue)
335  {
336  if (_numDefinitionColors < 1)
337  throw new InvalidOperationException("No definition colors are specified.");
338  else if (_numDefinitionColors == 1)
339  {
340  return _definitionColors[0];
341  } else if (_numDefinitionColors == 2)
342  {
343  return color.Mixture(true /* clipComponents */, 1.0*(1.0-referenceValue), _definitionColors[0],
344  1.0*referenceValue, _definitionColors[1]);
345  }
346  else if (_numDefinitionColors == 3)
347  {
348  if (referenceValue<=0.5)
349  return color.Mixture(true /* clipComponents */,
350  (1.0-2.0*referenceValue), _definitionColors[0],
351  2*referenceValue, _definitionColors[1]);
352  else
353  return color.Mixture(true /* clipComponents */,
354  (1.0-2.0*(referenceValue-0.5)), _definitionColors[1],
355  2.0*(referenceValue-0.5), _definitionColors[2]);
356  } else
357  {
358  // Remark: for smaller numvers of definiton colors the color is caluculated with special
359  // procedures for faster calculation. General method called below is a bit slower.
360  return color.Scale(_definitionColors, referenceValue);
361  }
362  }

376  public override double GetReferenceCellValue(int cellIndex)
377  {
378  if (_numCells <= 0)
379  throw new InvalidOperationException("Cell indices are not defined because number of cells is less than 1.");
380  if (cellIndex < 0)
381  return 0;
382  else if (cellIndex >= _numCells)
383  return 1.0;
384  else
385  {
386  double _referenceCellInterval = 1.0 / NumCells;
387  return ((double)cellIndex + 0.5) * _referenceCellInterval;
388  }
389  }

396  public override int GetReferenceCellIndex(double referenceValue)
397  {
398  if (referenceValue < 0)
399  return -1;
400  else if (referenceValue > 1)
401  return NumCells;
402  else
403  {
404  int ret = (int)Math.Floor(referenceValue * (double) _numCells);
405  // correct for possible roundoff errors:
406  if (ret == NumCells && referenceValue <= 1)
407  ret = _numCells - 1;
408  return ret;
409  }
410  }
