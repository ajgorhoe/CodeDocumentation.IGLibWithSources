534  public static string XPath(XmlNode node, XmlNode basenode)
535  {
536  try
537  {
538  if (node == null)
539  throw new ArgumentException("Xml.XPath(): The node whose XPath is queried is not specified (null reference).");
540  if (node == basenode)
541  return "/";
542  XmlNode
543  current = basenode,
544  parent = basenode.ParentNode; // parent of the current node
545  string ret = null;
546  if (parent == null)
547  ret = "/";
548  else
549  {
550  ret = "";
551  if (current.NodeType == XmlNodeType.Attribute)
552  {
553  // If the node is an attribute then add @ before its name:
554  if (current == basenode)
555  throw new ArgumentException("An attribute node is specified as the base node in XPath query.");
556  ret = "/@" + current.Name;
557  // move one messagelevel higher:
558  current = parent;
559  parent = current.ParentNode;
560  }
561  if (current != null)
562  {
563  int index;
564  do
565  {
566  // Calculate index of the current node in the list of all its siblings with the same name and
567  // create the portion of xpath string corresponding to this node:
568  string name = null;
569  try { name = current.Name; }
570  catch { }
571  index = ChildIndex(current);
572  if (string.IsNullOrEmpty(name))
573  {
574  ret += "[" + index + "]"; // current element has no name (e.g. comment or tect node), add to
575  // xpath only index of the element in square brackets, which will follow its parent name;
576  }
577  else
578  {
579  if (index == FirstChildIndex) // we do not need to state index of the current element if it is the first sibling with specific name
580  ret += "/" + current.Name;
581  else
582  ret += "/" + current.Name + "[" + index + "]";
583  }
584  // move one messagelevel higher:
585  current = parent;
586  parent = current.ParentNode;
587 
588  } while (parent != null);
589  }
590  }
591  }
