456  public bool WaitAllJobsCompleted(double timeoutInSeconds)
457  {
458  if (_outputLevel >= 1)
459  {
460  Console.WriteLine("Waiting job completion, timeout = " + timeoutInSeconds + " s...");
461  }
462  if (NumUncompletedJobs <= 0)
463  {
464  if (NumUncompletedJobs < 0)
465  throw new InvalidOperationException("Job dispatcher No. " + Id + ": number of executing jobs is less than 0.");
466  return true;
467  } else
468  {
469  bool completed = false, timeoutOccurred = false;
470  int sleepMs = SleepTimeMs, timeoutMs = (int)Math.Round(1e3 * timeoutInSeconds);
471  if (timeoutMs == 0)
472  {
473  if (timeoutInSeconds > 0) // prevent setting to 0 due to round-off
474  timeoutMs = 1;
475  }
476  int totalSleepTime = 0;
477  do
478  {
479  Thread.Sleep(sleepMs);
480  totalSleepTime += sleepMs;
481  completed = (NumUncompletedJobs <= 0); // check whether all jobs have completed
482  if (!completed)
483  {
484  if (timeoutMs > 0 && totalSleepTime >= timeoutMs)
485  timeoutOccurred = true;
486  } else
487  {
488  if (NumUncompletedJobs < 0)
489  throw new InvalidOperationException("Job dispatcher No. " + Id + ": number of executing jobs is less than 0.");
490  }
491  } while (!completed && !timeoutOccurred);
492  if (OutputLevel >= 1)
493  {
494  Console.WriteLine("Waiting job completion finished, job ID = " + Id + ", completed = " + completed);
495  }
496  return completed;
497  }
498  }

694  protected void Serve()
695  {
696  try
697  {
698  if (_outputLevel >= 1)
699  {
700  Console.WriteLine("Dispatcher " + Id + ": server started.");
701  }
702  _isServerRunning = true;
703  bool doServe = true;
704  while (doServe)
705  {
706  try
707  {
708  JobContainerType jobData = null;
709  ParallelJobServerBase<JobContainerType> idleServer = null;
710  do
711  {
712  // Try to pick & run the next enqueued job:
713  jobData = null;
714  lock (_jobQueue)
715  {
716  if (_jobQueue.Count > 0)
717  {
718  idleServer = GetFirstIdleServer();
719  if (idleServer != null)
720  {
721  jobData = _jobQueue.Dequeue();
722  }
723  }
724  }
725  if (jobData != null)
726  {
727  if (_outputLevel >= 1)
728  {
729  Console.WriteLine("Dispatcher " + Id + ": job dequeued and sent to server... "
730  + Environment.NewLine + " job ID = " + jobData.Id
731  + Environment.NewLine + ", disp. job ID: " + jobData.DispatcherJobId
732  + Environment.NewLine + ", server ID: " + idleServer.Id);
733  }
734  idleServer.StartJob(jobData);
735  }
736  if (CommandStopServing)
737  {
738  doServe = false;
739  CommandStopServing = false;
740  }
741  } while (doServe && jobData != null); // && NumIdleJobServers > 0);
742  if (CommandStopServing)
743  {
744  doServe = false;
745  CommandStopServing = false;
746  }
747  if (doServe)
748  Thread.Sleep(SleepTimeMs);
749  }
750  catch {
751  if (_outputLevel >= 1)
752  {
753  Console.WriteLine(Environment.NewLine +
754  "WARNING: exception thrown in server thread! Dispatcher ID: " + Id + Environment.NewLine);
755  }
756  }
757  }
758  }
759  catch
760  {
761  // Remark: Thread abortion is already handled in TunJobInThread.
762  if (_outputLevel >= 1)
763  {
764  Console.WriteLine("Dispatcher " + Id + ": WARNING: queue server stopped due to exception.");
765  }
766  }
767  finally
768  {
769  lock (Lock)
770  {
771  _isServerRunning = false;
772  _workingThread = null;
773  }
774  if (_outputLevel >= 1)
775  {
776  Console.WriteLine("Dispatcher " + Id + ": queue server stopped.");
777  }
778  }
