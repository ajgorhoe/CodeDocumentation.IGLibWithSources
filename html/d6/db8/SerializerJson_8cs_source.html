80  public override void Serialize<T>(T serializedObject, Stream serializationStream)
81  {
82  if (!Formatted)
83  base.Serialize<T>(serializedObject, serializationStream);
84  else
85  {
86  // We want to format serialized JSON text. First we create an unformatted string representation
87  // and then format it.
88  string str;
89  using (MemoryStream ms = new MemoryStream())
90  {
91  base.Serialize<T>(serializedObject, ms);
92  str = Encoding.Default.GetString(ms.ToArray());
93  // ms.Dispose(); - using instead.
94  }
95  string outStr = Formatter.Format(str);
96  using (StreamWriter sw = new StreamWriter(serializationStream))
97  {
98  sw.Write(outStr);
99  }
100  }
101  }
