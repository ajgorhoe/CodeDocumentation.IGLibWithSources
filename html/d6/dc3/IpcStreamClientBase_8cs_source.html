149  protected virtual string ReadResponse()
150  {
151  lock (Lock)
152  {
153  if (OutputLevel >= 1)
154  Console.WriteLine("Waiting for server response...");
155 
156  try
157  {
158  string serverResponseString = null;
159  if (IsMultilineResponse || true)
160  {
161  StringBuilderInternal.Clear();
162  bool stop = false;
163  bool responseBegun = false;
164  if (!IsMultilineResponse)
165  responseBegun = false;
166 
167  string line;
168  bool isMessage;
169  string messageOrCommandName;
170  string [] messageArguments;
171 
172  do
173  {
174  if (OutputLevel >= 2)
175  Console.WriteLine(" Waiting for next line of response...");
176  line = InputStream.ReadLine();
177  InterpretRequestOrResponseLine(ref line, out isMessage,
178  out messageOrCommandName, out messageArguments);
179 
180 
181  if (OutputLevel >= 2)
182  Console.WriteLine(" ... next line received: \"" + line + "\"");
183  if (!isMessage)
184  {
185  if (responseBegun)
186  {
187  StringBuilderInternal.AppendLine(line);
188  }
189  else
190  {
191  }
192  } else
193  {
194  // Message received:
195  bool messageWorked = false;
196 
197  if (messageOrCommandName == MsgResponseBegin)
198  {
199  responseBegun = true;
200  messageWorked = true;
201  if (!IsMultilineResponse)
202  throw new InvalidOperationException("Response begin message received in single line response mode.");
203  }
204  else if (messageOrCommandName == MsgResponseEnd)
205  {
206  stop = true;
207  messageWorked = true;
208  if (!IsMultilineResponse)
209  throw new InvalidOperationException("Response begin message received in single line response mode.");
210  } else
211  {
212  WorkMessage(messageOrCommandName, messageArguments, IpcStage.ReadingResponse, ref messageWorked);
213  if (!messageWorked)
214  throw new InvalidOperationException("Message not worked while working on server response: '" + messageOrCommandName + "'.");
215  }
216 
217  }

219

222  //else
223  // StringBuilderInternal.AppendLine(line);
224  } while (!stop);
225  serverResponseString = StringBuilderInternal.ToString();
226  serverResponseString = serverResponseString.TrimEnd('\n', '\r');
227  if (OutputLevel >= 1)
228  Console.WriteLine("Received multiline response: \"" + serverResponseString + "\"");
229  }
230 
231  else if (false)
232  {
233  if (OutputLevel >= 1)
234  {
235  Console.WriteLine("Waiting for single line response...");
236  }
237  serverResponseString = InputStream.ReadLine();
238  if (OutputLevel >= 1)
239  {
240  Console.WriteLine("... Single line response received: \"" + serverResponseString + "\"");
241  }
242  }
243  if (IsErrorResponse(serverResponseString))
244  {
245  // Error occurred on the server:
246  IsResponseReceived = true;
247  IsError = true;
248  LastException = null;
249  LastErrorMessage = GetErrorMessage(serverResponseString);
250  ResponseString = null;
251  if (OutputLevel >= 1)
252  {
253  Console.WriteLine("Error response obtained, message: \"" + LastErrorMessage + "\"" + Environment.NewLine);
254  }
255  }
256  else
257  {
258  IsResponseReceived = true;
259  IsError = false;
260  LastException = null;
261  LastErrorMessage = null;
262  ResponseString = serverResponseString;
263  if (OutputLevel >= 1)
264  {
265  Console.WriteLine("Response from server: \"" + serverResponseString + "\"");
266  }
267  }
268  LastRequestString = _requestString;
269  LastResponseString = _responseString;
270  if (IsError)
271  {
272  if (OutputLevel >= 1)
273  {
274  Console.WriteLine(Environment.NewLine + "Error occurred: "

276  }
277  }

283  }
284  }
285  }

290  LastRequestString = null;

292  LastException = ex;
