378  protected override void CopyToPlain(ref AnalysisResults anres)
379  {
380  base.CopyToPlain(ref anres);
381  if (anres!=null)
382  {
383  anres.ErrorCode = this.ErrorCode;
384  anres.ErrorString = this.ErrorString;
385 
386  anres.CalculatedObjective = this.CalculatedObjective;
387  anres.CalculatedConstraints = this.CalculatedConstraints;
388  anres.CalculatedObjectiveGradient = this.CalculatedObjectiveGradient;
389  anres.CalculatedConstraintGradients = this.CalculatedConstraintGradients;
390  anres.CalculatedObjectiveHessian = this.CalculatedObjectiveHessian;
391  anres.CalculatedConstraintHessians = this.CalculatedConstraintHessians;
392 
393  anres.Objective = Objective;
394  if (this.Constraints==null)
395  anres.Constraints = null;
396  else
397  {
398  int num = this.Constraints.Length;
399  List<double> constr = anres.Constraints;
400  Util.ResizeList<double>(ref constr, num, 0.0, true);
401  for (int i = 0; i < num; ++i)
402  constr[i] = this.Constraints[i];
403  anres.Constraints = constr;
404  }
405  if (this.ObjectiveGradient == null)
406  anres.ObjectiveGradient = null;
407  else
408  {
409  IVector grad = anres.ObjectiveGradient;
410  this.ObjectiveGradient.CopyTo(ref grad);
411  anres.ObjectiveGradient = grad;
412  }
413  if (this.ConstraintGradients == null)
414  anres.ConstraintGradients = null;
415  else
416  {
417  int num = this.ConstraintGradients.Length;
418  List<IVector> anresGradients = anres.ConstraintGradients;
419  Util.ResizeList<IVector>(ref anresGradients, num, null, true);
420  for (int i = 0; i < num; ++i)
421  {
422  if (this.ConstraintGradients[i] == null)
423  anresGradients[i] = null;
424  else
425  {
426  IVector grad = anresGradients[i];
427  this.ConstraintGradients[i].CopyTo(ref grad);
428  anresGradients[i] = grad;
429  }
430  }
431  anres.ConstraintGradients = anresGradients;
432  }
433 
434  if (this.ObjectiveHessian == null)
435  anres.ObjectiveHessian = null;
436  else
437  {
438  IMatrix hess = anres.ObjectiveHessian;
439  this.ObjectiveHessian.CopyTo(ref hess);
440  anres.ObjectiveHessian = hess;
441  }
442  if (this.ConstraintHessians == null)
443  anres.ConstraintHessians = null;
444  else
445  {
446  int num = this.ConstraintHessians.Length;
447  List<IMatrix> anresHessians = anres.ConstraintHessians;
