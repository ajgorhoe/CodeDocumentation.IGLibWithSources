652  public static void Example(string scriptPath)
653  {
654  Console.WriteLine();
655  Console.WriteLine("Example of dynamic function definitions.");
656  bool continueExample = true;
657  int numIterations = 0;
658  string valueDefinition = null, derivativeDefinition = null;
659  RealFunctionLoader loader = new RealFunctionLoader(); // create a new function loader
660  loader.IndependentVariableName = "x"; // name of independent variable used in expressions that define the function
661  StopWatch1 t = new StopWatch1();
662  while (continueExample)
663  {
664  try
665  {
666  Console.WriteLine();
667  loader.InvalidateDefinitions();
668  if (numIterations == 0)
669  {
670  valueDefinition = "x*x"; // expression for function value
671  derivativeDefinition = "2*x"; // expression for function derivative
672  }
673  else if (numIterations == 1)
674  {
675  valueDefinition = "pow(x,3)";
676  derivativeDefinition = "3*pow(x,2)";
677  }
678  else
679  {
680  Console.WriteLine("Input definition of function and (optionally) its derivative!");
681  Console.Write("f(" + loader.IndependentVariableName + ") = ");
682  valueDefinition = Console.ReadLine();
683  Console.Write("f'(" + loader.IndependentVariableName + ") = ");
684  derivativeDefinition = Console.ReadLine();
685  }
686  t.Start();
687  // Define the function that will be dynamically loaded:
688  loader.InvalidateDefinitions();
689  loader.ValueDefinitionString = valueDefinition;
690  loader.DerivativeDefinitionString = derivativeDefinition;
691  if (!string.IsNullOrEmpty(scriptPath))
692  {
693  File.WriteAllText(scriptPath, loader.Code);
694  Console.WriteLine("Script saved to " + scriptPath + ".");
695  }
696  // Create the corresponding function object:
697  IRealFunction func = loader.CreateRealFunction();
698  t.Stop();
699  Console.WriteLine("Function object generated in " + t.Time + " seconds.");
700  Console.WriteLine("Definition: ");
701  Console.WriteLine("f(" + loader.IndependentVariableName + ") = " + loader.ValueDefinitionString);
702  Console.WriteLine("f'(" + loader.IndependentVariableName + ") = " + loader.DerivativeDefinitionString);
703  // Use function object:
704  if (!func.ValueDefined)
705  Console.WriteLine("Function value is not defined!");
706  else
707  {
708  Console.WriteLine("Values: ");
709  for (int i = 0; i <= 5; ++i)
710  {
711  Console.Write("f(" + i + ") = " + func.Value(i) + ". ");
712  if (func.DerivativeDefined)
713  Console.Write("f'(" + i + ") = " + func.Derivative(i) + ". ");
714  Console.WriteLine();
715  }
716  }
717  }
718  catch (Exception ex)
719  {
720  Console.WriteLine("ERROR: " + ex.Message);
721  }
722  ++numIterations;
723  Console.WriteLine();
724  Console.Write("Continue example (0/1)? ");
725  UtilConsole.Read(ref continueExample);
726  }
727  Console.WriteLine();
728  }
