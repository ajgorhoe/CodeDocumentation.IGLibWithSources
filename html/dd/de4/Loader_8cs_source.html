193  public static ILoadableScript CreateScriptObject(string scriptClassFullName, int outputLevel)
194  {
195  ILoadableScript script = null;
196  try
197  {
198  // Try to find and instantiate the class in the process' entry assembly:
199  Assembly assembly = Assembly.GetEntryAssembly();
200  script = GetLoadableScriptC((ILoadableScript)assembly.CreateInstance(scriptClassFullName));
201  if (script!=null && outputLevel>=1)
202  Console.WriteLine("Executing script " + script.GetType().FullName + " from entry assembly " + assembly.FullName +
203  "..." + Environment.NewLine);
204  }
205  catch { }
206  if (script == null)
207  {
208  // The specified class does not exist in the current assembly. Try to find it in all loaded assemblies:
209  foreach (Assembly assembly in AppDomain.CurrentDomain.GetAssemblies())
210  {
211  try
212  {
213  if (script == null)
214  {
215  script = GetLoadableScriptC((ILoadableScript)assembly.CreateInstance(scriptClassFullName));
216  if (script!=null && outputLevel>=1)
217  Console.WriteLine("Executing script " + script.GetType().FullName + " from loaded assembly " + assembly.FullName +
218  "..." + Environment.NewLine);
219  }
220  }
221  catch{ }
222  }
223  }
224  if (script == null)
225  {
226  // The specified class does not exist in the current assembly. Try to find it in all assemblies referenced
227  // by the entry assembly:
228  foreach (AssemblyName assemblyName in Assembly.GetEntryAssembly().GetReferencedAssemblies())
229  {
230  try
231  {
232  if (script == null)
233  {
234  Assembly assembly = Assembly.Load(assemblyName);
235  script = GetLoadableScriptC((ILoadableScript)assembly.CreateInstance(scriptClassFullName));
236  if (script!=null && outputLevel>=1)
237  Console.WriteLine("Executing script " + script.GetType().FullName + " from directly referenced assembly " + assembly.FullName +
238  "..." + Environment.NewLine);
239  }
240  }
241  catch{ }
242  }
243  }
244  if (script == null)
245  {
246  // The specified class does not exist in any assembly directly reeferenced by the entry assembly.
247  // Try to find it in all directly or indirectly referenced assemblies up to the first recursion level:
248  foreach (AssemblyName referencedAssemblyName in Assembly.GetEntryAssembly().GetReferencedAssemblies())
249  {
250  try
251  {
252  Assembly referencedAssembly = Assembly.Load(referencedAssemblyName);
253  foreach (AssemblyName assemblyName in referencedAssembly.GetReferencedAssemblies())
254  {
255  try
256  {
257  if (script == null)
258  {
259  Assembly assembly = Assembly.Load(assemblyName);
260  script = GetLoadableScriptC((ILoadableScript)assembly.CreateInstance(scriptClassFullName));
261  if (script != null && outputLevel >= 1)
262  Console.WriteLine("Executing script " + script.GetType().FullName + " from indirectly referenced assembly " + assembly.FullName +
263  "..." + Environment.NewLine);
264  }
265  }
266  catch { }
267  }
268  }
269  catch { }
270  }
271  }
272  if (script == null)
273  {
274  // The specified class was not find, try to find it again in all loaded assemblies:
275  foreach (Assembly assembly in AppDomain.CurrentDomain.GetAssemblies())
276  {
277  try
278  {
279  if (script == null)
280  {
281  script = GetLoadableScriptC((ILoadableScript)assembly.CreateInstance(scriptClassFullName));
282  if (script != null && outputLevel >= 1)
283  Console.WriteLine("Executing script " + script.GetType().FullName + " from indirectly referenced assembly " + assembly.FullName +
284  "..." + Environment.NewLine);
285  }
286  }
287  catch { }
288  }
289  }
290  script = GetLoadableScriptC(script);
291  return script;
292  }

533  public static string FindNameAfterKeyword(string code, string keyword, string allowedCharacters)
534  {
535  if (string.IsNullOrEmpty(code) || string.IsNullOrEmpty(keyword))
536  return null;
537  int pos, posAfter;
538  pos = code.IndexOf(keyword); // find the keyword
539  if (pos < 0)
540  return null;
541  pos += keyword.Length; // one after the keyword
542  int codeLength = code.Length;
543  while (pos < codeLength) // first non-blank after the keyword
544  {
545  char current = code[pos];
546  if (!char.IsWhiteSpace(current))
547  break;
548  ++pos;
549  }
550  if (pos >= codeLength)
551  return null;
552  // Find the first character after the name:
553  posAfter = pos+1;
554  while (posAfter < codeLength)
555  {
556  char current = code[posAfter];
557  bool isPartOfName = false;
558  if (char.IsLetterOrDigit(current))
559  isPartOfName = true;
560  //else if (current=='_')
561  // partOfName = true;
562  else
563  {
564  if (allowedCharacters!=null)
565  if (allowedCharacters.IndexOf(current) >= 0)
566  isPartOfName = true;
567  }
568  if (!isPartOfName)
569  {
570  break;
571  }
572  if (allowedCharacters!=null)
573 
574  ++posAfter;
575  }
576  return code.Substring(pos, posAfter - pos);
577  }
