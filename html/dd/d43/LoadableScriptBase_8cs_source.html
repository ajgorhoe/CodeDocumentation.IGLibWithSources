577  string commandName, Script_CommandDelegate command, string helpString)
578  {

695  public string Script_Run(string[] arguments)
696  {
697  if (arguments == null)
698  throw new ArgumentException("Arguments not specified (null array).");
699  else if (arguments.Length < 1)
700  throw new ArgumentException("Number of arguments less than 1. At least command name should be specified.");
701  string commandName = arguments[0];
702  if (string.IsNullOrEmpty(commandName))
703  throw new ArgumentException("Command name (the 0-th argument) is not specified (null or empty string).");
704  if (Script_ContainsCommand(commandName))
705  {
706  if (Script_ContainsScriptCommand(commandName))
707  {
708  // We are running a script command, therefore it is run without modifications:
709  return Script_Interpreter.RunWithoutModifications(Script_Interpreter.MainThread, commandName, arguments);
710  }
711  else
712  {
713  string[] args = null;
714  // We are running a normal interpreter command through Script_Run, we therefore have to take away the first
715  // argument that is the command name:
716  if (arguments != null)
717  {
718  int numArgs = arguments.Length - 1;
719  if (numArgs < 0)
720  numArgs = 0;
721  args = new string[numArgs];
722  for (int i = 0; i < numArgs; ++i)
723  args[i] = arguments[i + 1];
724  }
725  return Script_Interpreter.RunWithoutModifications(Script_Interpreter.MainThread, commandName, args);
726  }
727 
728  } else
729  throw new InvalidOperationException("Script: unknown internal or interpreter command " + commandName + ".");
730  }
