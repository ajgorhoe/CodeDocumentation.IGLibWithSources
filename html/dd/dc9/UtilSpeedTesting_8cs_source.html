267  public static double StandardSpeedTestGeometricSeries_HpLaptop24(out SpeedTestInfo info,
268  bool writeToConsole = DefaultWriteToConsole_StandardTests)
269  {
270  info = null;
271  try
272  {
273  double a0 = ConstGeom.a0;
274  double k = ConstGeom.k;
275  info = new(ConstGeom.TestId, ConstMachineHpLaptop24.MachineId);
276  info.Parameters.AddRange([("a0", a0), ("k", k)]);
277  info.NumExecutions = ConstGeom.NumExecutions;
278  info.AnalyticalResult = ConstGeom.AnalyticResult;
279  info.Tolerance = ConstGeom.Tolerance;
280  info.ReferenceExecutionsPerSecond = ConstGeom.RefExecutionsPerSecond_HpLaptop24;
281  if (writeToConsole)
282  {
283  Console.WriteLine($"Performing speed test - numerical calculation of a finite geometric series...");
284  Console.WriteLine($"Test ID: {ConstGeom.TestId}");
285  Console.WriteLine($"Parameters:");
286  Console.WriteLine($" n: {info.NumExecutions} (number of iterations)");
287  Console.WriteLine($" a: {a0} (the first element)");
288  Console.WriteLine($" k: {k} (quotient of successive elements)");
289  Console.WriteLine($" tol: {info.Tolerance} (tolerance (max. allowed discrepancy))");
290  Console.WriteLine($" k^n: {Math.Pow(k, info.NumExecutions)}");
291  }
292  Stopwatch sw = Stopwatch.StartNew();
293  try
294  {
295  info.Result = GeometricSeriesNumerical(info.NumExecutions, a0, k);
296  info.ExecutionTimeSeconds = sw.Elapsed.TotalSeconds;
297  }
298  finally
299  {
300  sw.Stop();
301  }
302  if (writeToConsole)
303  {
304  Console.WriteLine($"Numerical result: S = {info.Result}");
305  Console.WriteLine($"Analytical result: S = {info.AnalyticalResult}");
306  Console.WriteLine($" Difference: {info.Result - info.AnalyticalResult}, tolerance: {info.Tolerance}");
307  }
308  // Verification - numerical result needs to be correct within the specified tolerance:
309  if (Math.Abs(info.Result - info.AnalyticalResult) > info.Tolerance)
310  {
311  if (writeToConsole)
312  {
313  Console.WriteLine("WARNING: Diecrepancy between the numerical and analytical result is not within tolerance; exception is being thrown.");
314  throw new InvalidOperationException("Discrepancy between numerical and analytical result is not within tolerance.");
315  }
316  }
317  if (writeToConsole)
318  {
319  Console.WriteLine("");
320  Console.WriteLine($"Geometric series with {info.NumExecutions} terms was calculated in {info.ExecutionTimeSeconds} s.");
321  Console.WriteLine($" Calculations per second: {info.NumExecutionsPerSecond}");
322  Console.WriteLine($" Millions per second: {info.MegaExecutionsPerSecond}");
323  Console.WriteLine($" Speed factor: {info.SpeedFactor}");
324  }
325  }
326  catch (Exception ex)
327  {
328  if (info != null) { info.Exception = ex; }
329  throw;
330  }
331  if (info != null) { return info.SpeedFactor; }
332  return default;
