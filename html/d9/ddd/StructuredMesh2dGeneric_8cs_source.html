195  public void AddIndexField(TIndexField field)
196  {
197  if (field == null)
198  throw new ArgumentNullException("field", "Field to be added is not specified (null argument).");
199  if (field.Dim1 != this.Dim1 || field.Dim2 != this.Dim2)
200  {
201  if (field.Length == 0)
202  field.SetDimensions(this.Dim1, this.Dim2);
203  else
204  throw new ArgumentException("Dimensions of the field to be added (" + field.Dim1 + ", " + field.Dim2 +
205  ") do not match dimension of the mesh (" + this.Dim1 + ", " + this.Dim2 + ").");
206  }
207  _activeIndexField = null; // since this might be dependent on the performed operation
208  IndexFields.AddField(field);
