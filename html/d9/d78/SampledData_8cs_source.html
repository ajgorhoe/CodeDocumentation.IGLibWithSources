565  public void ExtractFilteredData(IndexList filterIndices, ref IVector[] extracted,
566  bool complement, bool outputData, bool copyValues)
567  {
568  int numElements = this.Length;
569  int numFilterIndices = 0; // number of indices in the filter
570  if (filterIndices != null)
571  numFilterIndices = filterIndices.Length;
572  int numToExtract; // number of vectors to extract
573  if (complement)
574  numToExtract = numElements - numFilterIndices;
575  else
576  numToExtract = numFilterIndices;
577  int whichElement = 0; // current element of this data set
578  int whichExtracted = 0; // current element of extracted vectors array
579  int whichFilterIndex = 0; // current element in filter index list
580  if (extracted == null)
581  extracted = new IVector[numToExtract];
582  else if (extracted.Length != numToExtract)
583  extracted = new IVector[numToExtract];
584  while (whichExtracted < numToExtract)
585  {
586  bool extractThis; // whether to extract from the current element
587  if (complement)
588  {
589  // Extracted data is from those elements whose indices are NOT contained in filterIndices
590  extractThis = true; // this holds in the case that filter indices do not contain index of the current element
591  if (whichFilterIndex < numFilterIndices)
592  if (filterIndices[whichFilterIndex] == whichElement)
593  {
594  extractThis = false;
595  ++whichFilterIndex;
596  }
597  }
598  else
599  {
600  // Extracted data is from those elements whose indices ARE contained in filterIndices
601  extractThis = false; // this holds in the case that filter indices do not contain index of the current element
602  if (whichFilterIndex < numFilterIndices)
603  if (filterIndices[whichFilterIndex] == whichElement)
604  {
605  extractThis = true;
606  ++whichFilterIndex;
607  }
608  }
609  if (extractThis)
610  {
611  if (whichElement >= numElements)
612  {
613  throw new InvalidOperationException("UnExpectedly, no elements in the samped data set left for extraction." + Environment.NewLine
614  + " Number of elements: " + numElements + ", current: " + whichElement + Environment.NewLine
615  + " Number of indices: " + numFilterIndices + ", current: " + whichFilterIndex + Environment.NewLine
616  + " Number to extract: " + numToExtract + ", current: " + whichExtracted);
617  }
618  IVector extractedVector;
619  if (outputData)
620  extractedVector = GetOutputValues(whichElement);
621  else
622  extractedVector = GetInputParameters(whichElement);
623  if (copyValues)
624  {
625  Vector.Copy(extractedVector, ref extracted[whichExtracted]);
626  }
627  else
628  extracted[whichExtracted] = extractedVector;
629  ++whichExtracted;
630  ++whichElement;
631  }
632  }
