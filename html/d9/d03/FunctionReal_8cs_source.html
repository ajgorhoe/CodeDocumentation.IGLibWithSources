301  public void Tabulate(double from, double to, int numPoints,
302  bool printDerivatives, bool printSecondDerivatives)
303  {
304  if (!this.ValueDefined)
305  throw new ArgumentException("Can not tabulate the function, value is not defined.");
306  Console.WriteLine("Tabulating function, {0} values from {0,G5} to {1,G5}");
307  if (printDerivatives)
308  {
309  if (!this.DerivativeDefined)
310  {
311  printDerivatives = false;
312  Console.WriteLine("Derivatives will not be printed; not defined.");
313  }
314  }
315  if (printSecondDerivatives)
316  {
317  if (!this.SecondDerivativeDefined)
318  {
319  printSecondDerivatives = false;
320  Console.WriteLine("Second derivatives will not be printed; not defined.");
321  }
322  }
323  Console.WriteLine("Function: " + this.ToString());
324  Console.Write(" {0,10} {1,-10}", "x", "f(x)");
325  if (printDerivatives)
326  Console.Write(" {0,-10}", "f'(x)");
327  if (printSecondDerivatives)
328  Console.Write(" {0,-10}", "f''(x)");
329  Console.WriteLine();
330  double step = to - from;
331  if (numPoints > 2)
332  step /= (double)(numPoints - 1);
333  for (int i = 0; i < numPoints; ++i)
334  {
335  double x = from + (double)i * step;
336  double f = this.Value(x);
337  Console.Write(" {0,10:G8} {1,-10:G8}", x, f);
338  if (printDerivatives)
339  {
340  double fDer = this.Derivative(x);
341  Console.Write(" {0,-10:G8}", fDer);
342  }
343  if (printSecondDerivatives)
344  {
345  double fDer2 = this.SecondDerivative(x);
346  Console.Write(" {0,-10:G8}", fDer2);
347  }
348  Console.WriteLine();
349  }

405  public virtual void Test(double from, double to, int numProbes, double stepSize, double tolerance)
406  {
407  Console.WriteLine();
408  Console.WriteLine("Testing correctness of implementation of function " + Name + ".");
409  Console.WriteLine("Test intervel: [" + from + ", " + to + "] , num. probes: "
410  + numProbes + ", step: " + stepSize + ", tolerance: ", + tolerance);
411  double x;
412  double error, relativeerror;
413  if (tolerance<0)
414  tolerance = -tolerance;
415  double probeStep = (to-from)/((double) numProbes - 1.0);
416  // First derivative:
417  if (DerivativeDefined)
418  {
419  Console.WriteLine();
420  Console.WriteLine("Derivative:");
421  double der, numder;
422  for (int i=0; i<numProbes; ++i)
423  {
424  x = from + (double)i * probeStep;
425  der = Derivative(x);
426  numder = NumericalDerivative(x, stepSize);
427  error = numder - der;
428  relativeerror = Math.Abs(2 * (numder - der) / (Math.Abs(numder) + Math.Abs(der)));
429  if (Math.Abs(error) > tolerance)
430  Console.WriteLine("!!!! large ERROR:");
431  Console.WriteLine("x: " + x.ToString() + ", der.: " + der + ", " + Environment.NewLine +
432  " err: " + error + ", rel.: " + relativeerror );
433  }
434  } else
435  Console.WriteLine("Analytical derivative is not defined." + Environment.NewLine);
436  // Second derivative:
437  if (SecondDerivativeDefined)
438  {
439  Console.WriteLine();
440  Console.WriteLine("Second derivative:");
441  double der, numder;
442  for (int i=0; i<numProbes; ++i)
443  {
444  x = from + (double)i * probeStep;
445  der = SecondDerivative(x);
446  numder = NumericalSecondDerivative(x, stepSize);
447  error = numder - der;
448  relativeerror = Math.Abs(2 * (numder - der) / (Math.Abs(numder) + Math.Abs(der)));
449  if (Math.Abs(error) > tolerance)
450  Console.WriteLine("!!!! large ERROR:");
451  Console.WriteLine("x: " + x.ToString() + ", der.: " + der + ", " + Environment.NewLine +
452  " err: " + error + ", rel.: " + relativeerror);
453  }
454  } else
455  Console.WriteLine("Analytical second derivative is not defined." + Environment.NewLine);
456 
457  // Integral:
458  if (IntegralDefined)
459  {
460  Console.WriteLine();
461  Console.WriteLine("Integral:");
462  double integral, numintegral;
463  for (int i = 0; i < numProbes; ++i)
464  {
465  x = from + (double)i * probeStep;
466  integral = Integral(x);
467  numintegral = NumericalIntegral(0, x, numProbes);
468  error = numintegral - integral;
469  relativeerror = Math.Abs(2 * (numintegral - integral) / (Math.Abs(numintegral) + Math.Abs(integral)));
470  if (Math.Abs(error) > tolerance)
471  Console.WriteLine("!!!! large ERROR:");
472  Console.WriteLine("x: " + x.ToString() + ", int.: " + integral + ", " + Environment.NewLine +
473  " err: " + error + ", rel.: " + relativeerror);
474  }
475  } else
476  Console.WriteLine("Analytical integral is not defined." + Environment.NewLine);
477 
478  // Inverse:
479  if (InverseDefined)
480  {
481  Console.WriteLine();
482  Console.WriteLine("Inverse:");
483  double y;
484  double xinv;
485  double yinv;
486  for (int i = 0; i < numProbes; ++i)
487  {
488  x = from + (double)i * probeStep;
489  y = Value(x);
490  xinv = Inverse(y);
491  yinv = Value(xinv);
492  error = yinv - y;
493  relativeerror = Math.Abs(2 * (yinv - y) / (Math.Abs(yinv) + Math.Abs(y)));
494  if (Math.Abs(error) > tolerance)
495  Console.WriteLine("!!!! large ERROR:");
496  Console.WriteLine("x: " + x.ToString() + ", x inv.: " + xinv + ", " + Environment.NewLine +
497  " err: " + error + ", rel.: " + relativeerror);
498  }
499  } else
500  Console.WriteLine("Analytical inverse is not defined." + Environment.NewLine);
501  Console.WriteLine();
502  }
