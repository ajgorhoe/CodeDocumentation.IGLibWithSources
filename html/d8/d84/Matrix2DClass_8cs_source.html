99  public Matrix2d(double[][] A)
100  {
101  if (A == null)
102  throw new ArgumentNullException("Jagged array to create a 2D matrix from is not specified (null reference).");
103  if (A.Rank != 2)
104  throw new ArgumentException("Jagged array to create a 2D matrix from is not a rank 2 array.");
105  if (A.Length != 2)
106  throw new ArgumentException("Jagged array to create a 2D matrix from does not have 2 rows.");
107  for (int i = 0; i < 2; ++i)
108  {
109  if (A[i].Length != 2)
110  throw new ArgumentException("Jagged array to create a 2D matrix from has inconsistent dimensions. Row "
111  + i + "has length " + A[i].Length + "Instead of 2.");
112  for (int j = 0; j < 2; ++j)
113  this[i, j] = A[i][j];
114  }
115  }

168  public static Matrix2d CreateFromColumns(IList<IVector> columnVectors)
169  {
170  if (columnVectors == null)
171  throw new ArgumentNullException("List of column vectors to create a 2D matrix from is not specified (null reference).");
172  if (columnVectors.Count != 2)
173  throw new ArgumentException("List of column vectors to create a 2D matrix from does not contain 2 vectors.");
174  int rows = columnVectors[0].Length;
175  int columns = columnVectors.Count;
176  if (rows!=2)
177  throw new ArgumentException("Length of column vectors to create a 2D matrix from is not 2.");
178  double[,] newData = new double[2,2];
179  for (int i = 0; i < rows; i++)
180  for (int j = 0; j < columns; j++)
181  newData[i,j] = columnVectors[j][i];
182  return new Matrix2d(newData);
183  }

186  public static Matrix2d CreateFromRows(IList<Vector> rowVectors)
187  {
188  if (rowVectors == null)
189  throw new ArgumentNullException("List of row vectors to create a 2D matrix from is not specified (null reference).");
190  int rows = rowVectors.Count;
191  int columns = rowVectors[0].Length;
192  if (rows != 2)
193  throw new ArgumentException("List of row vectors to create a 2D matrix from does not contain 2 vectors.");
194  if (columns != 2)
195  throw new ArgumentException("Length of row vectors to create a 2D matrix from is not 2.");
196  double[][] newData = new double[rows][];
197 
198  for (int i = 0; i < rows; i++)
199  {
200  newData[i] = rowVectors[i].ToArray();
201  }
202  return new Matrix2d(newData);
