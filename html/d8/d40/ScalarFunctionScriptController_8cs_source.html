425  public static void CreateParameterNamesArray(string namesString, string[] separators, ref string[] namesArray,
426  bool trimSpaces = true, bool allowMultipleSeparators = false)
427  {
428  int num = 0;
429  if (string.IsNullOrEmpty(namesString))
430  {
431  if (namesArray != null)
432  if (namesArray.Length > 0)
433  namesArray = null;
434  } else
435  {
436  string[] names = namesString.Split(separators, StringSplitOptions.None);
437  for (int i = 0; i < names.Length; ++i)
438  {
439  if (!string.IsNullOrEmpty(names[i]))
440  names[i] = names[i].Trim();
441  if (!string.IsNullOrEmpty(names[i]))
442  {
443  names[num] = names[i];
444  ++num;
445  }
446  }
447  if (num < 1)
448  {
449  if (namesArray != null)
450  if (namesArray.Length > 0)
451  namesArray = null;
452  } else
453  {
454  // There is at least one parameter name.
455  if (namesArray != null)
456  {
457  if (namesArray.Length != num)
458  {
459  namesArray = null;
460  }
461  }
462  if (namesArray == null)
463  {
464  namesArray = new string[num];
465  }
466  for (int i = 0; i < num; ++i)
467  {
468  namesArray[i] = names[i];
469  }
470  }
471  }

795  public virtual InputOutputDataDefiniton GetDataDefinitionObject(bool setBoundsAndDefaults = false, IBoundingBox inputBounds = null, IBoundingBox outputBounds = null)
796  {
797  InputOutputDataDefiniton ret = new InputOutputDataDefiniton();
798  // string digitFormatString = new string('0', numNameDigits);
799  for (int i = 0; i < Dimension; ++i)
800  {
801  InputElementDefinition inputElement = new InputElementDefinition(i, null, null);
802  inputElement.Name = inputElement.Title = ParameterNames[i];
803  inputElement.Description = "Function parameter " + i + "(" + ParameterNames[i] + ").";
804  if (setBoundsAndDefaults)
805  {
806  inputElement.MinimalValue = _defaultLowerBound;
807  inputElement.MaximalValue = _defaultUpperBound;
808  inputElement.DefaultValue = _defaultLowerBound;
809  inputElement.BoundsDefined = true;
810  inputElement.DefaultValueDefined = true;
811  if (inputBounds != null)
812  {
813  if (inputBounds.IsMinDefined(i))
814  inputElement.MinimalValue = inputBounds.GetMin(i);
815  else
816  inputElement.MinimalValue = double.MinValue;
817  if (inputBounds.IsMaxDefined(i))
818  inputElement.MaximalValue = inputBounds.GetMax(i);
819  else
820  inputElement.MaximalValue = double.MaxValue;
821  if (!(inputBounds.IsMinDefined(i) || inputBounds.IsMinDefined(i)))
822  inputElement.BoundsDefined = false;
823  if (inputElement.MinimalValue > inputElement.MaximalValue)
824  {
825  // swap lower and upper bound if necessary:
826  double min = inputElement.MinimalValue;
827  inputElement.MinimalValue = inputElement.MaximalValue;
828  inputElement.MaximalValue = min;
829  }
830  if (inputBounds.IsMinDefined(i) && inputBounds.IsMinDefined(i))
831  inputElement.DefaultValue = 0.5 * (inputElement.MinimalValue + inputElement.MaximalValue);
832  }
833  }
834  ret.AddInputElement(inputElement, true /* assign element index */);
835  }
836  for (int i = 0; i < 1; ++i)
837  {
838  OutputElementDefinition outputElement = new OutputElementDefinition(i, null, null);
839  outputElement.Name = outputElement.Title = Name;
840  outputElement.Description = "Scalar function \"" + Name + "\".";

844  outputElement.MaximalValue = _defaultUpperBound;
845  outputElement.BoundsDefined = true;
