477  public static int TestExecutionTime(VoidDelegate work, out double averageExecutionTime, out double averageCpuTime,
478  double targetedTime = 0.1, int outpuLevel = 0, int numInitialExecutions = 1)
479  {
480  int batchSize = numInitialExecutions;
481  if (batchSize < 1)
482  batchSize = 1;
483  IG.Lib.StopWatch1 t = new IG.Lib.StopWatch1();
484  int numExecutions = 0;
485  double totalTime = 0.0, totalCpuTime = 0;
486  double timeRatio = 0.0;
487  bool stop = false;
488  if (outpuLevel >= 1)
489  {
490  Console.WriteLine("Starting the time performance test...");
491  }
492  while (!stop)
493  {
494  t.Start();
495  for (int i = 0; i < batchSize; ++i)
496  {
497  work();
498  ++ numExecutions;
499  }
500  t.Stop();
501  totalTime += t.Time;
502  totalCpuTime += t.CpuTime;
503  timeRatio = totalTime / targetedTime;
504  if (outpuLevel >= 2)
505  {
506  Console.WriteLine(" " + batchSize + " executions performed in " + t.Time + " s (" + t.CpuTime + "CPU) " + Environment.NewLine
507  + " total time: " + totalTime + " s (CPU: " + totalCpuTime + "), time ratio: " + timeRatio );
508  }
509  if (timeRatio >= 1.0)
510  stop = true;
511  else if (timeRatio < 0.4)
512  {
513  if (timeRatio < 0.01)
514  {
515  if (numExecutions > 100)
516  batchSize = (int)(batchSize * 10);
517  else
518  batchSize *= 4;
519  } else if (timeRatio < 0.1)
520  {
521  if ((numExecutions > 20))
522  batchSize *= 6;
523  else
524  batchSize *= 3;
525  } else if (timeRatio < 0.1)
526  {
527  if (numExecutions > 5)
528  batchSize *= 4;
529  else
530  batchSize *= 2;
531  } else
532  {
533  if (numExecutions > 5)
534  {
535  batchSize = (int)(0.6 * (1.0 - timeRatio) * (double)numExecutions / (timeRatio));
536  if (batchSize < 1)
537  batchSize = 1;
538  }
539  else
540  {
541  batchSize = (int) (1.5 * (double) batchSize);
542  if (batchSize < 1)
543  batchSize = 1;
544  }
545  }
546  } else if (timeRatio < 0.8)
547  {
548  if (numExecutions > 5)
549  {
550  batchSize = (int)(0.8 * (1.0 - timeRatio) * (double)numExecutions / (timeRatio));
551  if (batchSize < 1)
552  batchSize = 1;
553  } else
554  {
555  batchSize = (int)(0.6 * (1.0 - timeRatio) * (double)numExecutions / (timeRatio));
556  if (batchSize < 1)
557  batchSize = 1;
558  }
559  } else if (timeRatio < 0.9)
560  {
561  batchSize = (int)(0.9 * (1.0 - timeRatio) * (double)numExecutions / (timeRatio));
562  if (batchSize < 1)
563  batchSize = 1;
564  } else if (timeRatio < 0.99)
565  {
566  batchSize = (int)(1.1 * (1.0 - timeRatio) * (double)numExecutions / (timeRatio));
567  if (batchSize < 1)
568  batchSize = 1;
569  } else
570  {
571  batchSize = (int)(1.5 * (1.0 - timeRatio) * (double)numExecutions / (timeRatio));
572  if (batchSize < 1)
573  batchSize = 1;
574  }
575  if (batchSize < 1)
576  batchSize = 1;
577  }
578  averageExecutionTime = totalTime / (double) numExecutions;
579  averageCpuTime = totalCpuTime / (double) numExecutions;
580  if (outpuLevel >=1)
581  {
582  Console.WriteLine(" ... test finished.");
583  Console.WriteLine(numExecutions + " executions performed in " + totalTime + " s (" + totalCpuTime + " s CPU).");
584  Console.WriteLine("Executions per second: " + ((double)numExecutions / totalTime) + " (per second of CPU time: "
585  + ((double)numExecutions / totalCpuTime) + ").");
586  Console.WriteLine("Average execution time: " + averageExecutionTime + " s (CPU: " + averageCpuTime + " s).");
587  }
588  return numExecutions;
589  }
