309  public override void Wait()
310  {
311  bool stop = false;
312  double totaltime = 0.0;
313  int sleeptime = 0;
314  bool doSleep = true;
315  try
316  {
317  lock (InternalLock)
318  {
319  IsWaiting = true;
320  CancelFlag = false;
321  Timer.Reset();
322  Timer.Start();
323  doSleep = (SleepFirst && MinSleepMs >= 0);
324  }
325  if (doSleep)
326  Thread.Sleep(MinSleepMs);
327  while (!stop)
328  {
329  if (Condition() || CancelFlag)
330  {
331  lock (InternalLock)
332  {
333  stop = true;
334  if (CancelFlag)
335  CancelFlag = false;
336  }
337  }
338  else
339  {
340  lock (InternalLock)
341  {
342  // Calculate the appropriate sleeping time and sleep before the next evaluation of
343  // unblocking condition:
344  sleeptime = 0;
345  if (MaxRelativeLatency > 0.0)
346  {
347  totaltime = Timer.Time; // total waiting time up to now, in seconds
348  sleeptime = (int)Math.Floor(1000 * totaltime * MaxRelativeLatency);
349  }
350  if (MinSleepMs >= 0 && sleeptime < MinSleepMs)
351  sleeptime = MinSleepMs;
352  if (MaxSleepMs >= 0 && sleeptime > MaxSleepMs)
353  sleeptime = MaxSleepMs;
354  }
355  if (sleeptime > 0)
356  Thread.Sleep(sleeptime);
357  }
358  } // lock InternalLock
359  }
360  catch
361  {
362  throw;
363  }
364  finally
365  {
366  IsWaiting = false;
367  }
368 
369  } // Wait()
