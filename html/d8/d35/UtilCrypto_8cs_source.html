533  public static void ParseHashFile(string filePath, ref List<string[]> hashList)
534  {
535  if (string.IsNullOrEmpty(filePath))
536  throw new ArgumentException("File path not specified (null or empty string).");
537  if (!File.Exists(filePath))
538  throw new ArgumentException("File containing hashes does not exist. " + Environment.NewLine
539  + " File path: " + filePath);
540  using (StreamReader reader = new StreamReader(filePath))
541  {
542  string line;
543  while ((line = reader.ReadLine()) != null)
544  {
545  line = line.Trim();
546  char[] chars = line.ToArray();
547  int length = chars.Length;
548  int firstSpace = -1;
549  int firstNonSpace = -1;
550  int pos = 0;
551  // Find the end of the hash string:
552  while (pos < length && firstSpace < 0)
553  {
554  if (char.IsWhiteSpace(chars[pos]))
555  firstSpace = pos;
556  ++pos;
557  }
558  if (firstSpace > 0)
559  {
560  while (pos < length && firstNonSpace < 0)
561  {
562  if (!char.IsWhiteSpace(chars[pos]))
563  firstNonSpace = pos;
564  ++pos;
565  }
566  }
567  if (firstSpace > 0 && firstNonSpace > firstSpace)
568  {
569  string hash = new string(chars, 0, firstSpace);
570  string hashedFile = new string(chars, firstNonSpace, length - firstNonSpace);
571  if (hashList == null)
572  hashList = new List<string[]>();
573  hashList.Add(new string[] { hash, hashedFile });
574  }
575  }
576  }
