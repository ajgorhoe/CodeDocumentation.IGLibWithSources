768  public static int EigenSystem2d(mat2 a, ref mat2 eigenvec, ref vec2 eigenval)
769  {
770  double D2 = a.xx * a.xx + 4 * a.xy * a.yx - 2 * a.xx * a.yy + a.yy * a.yy;
771  if (D2 < 0)
772  throw new ArgumentException("2D eigensystem has no real solutions.");
773  double D = Math.Sqrt(D2);
774  double norm;
775  eigenval.x = 0.5*(a.xx + a.yy - D);
776  eigenval.y = 0.5*(a.xx + a.yy + D);
777  // First eigenvector (stored as row)
778  eigenvec.xx = (a.xx-a.yy-D)/(2.0*a.yx);
779  eigenvec.xy=1;
780  norm = Math.Sqrt(eigenvec.xx*eigenvec.xx+eigenvec.xy*eigenvec.xy);
781  if (norm > 0)
782  {
783  eigenvec.xx /= norm;
784  eigenvec.xy /= norm;
785  }
786  // Second eigenvector (stored as row):
787  eigenvec.yx = (a.xx-a.yy+D)/(2.0*a.yx);
788  eigenvec.yy = 1;
789  norm = Math.Sqrt(eigenvec.yx*eigenvec.yx+eigenvec.yy*eigenvec.yy);
790  if (norm > 0)
791  {
792  eigenvec.yx /= norm;
793  eigenvec.yy /= norm;
794  }
795  return 2;
796  } // EigenSystem2d
