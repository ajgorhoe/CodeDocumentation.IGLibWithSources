280  public QuadraticBasis(int dimension)
281  {
282  Name = "QuadraticBasis" + dimension + "D";
283  Description = "Quadratic basis in " + dimension + " dimensions.";
284  int numFunc = (dimension+1) * (dimension+2) / 2;
285  NumParameters = dimension;
286  NumValues = numFunc;
287  // Compose lookup table of indices that enable quick querying of basis
288  // function definitions:
289  functionDefinitions = new int[numFunc][];
290  quatraticTerms = new int[numFunc, numFunc]; // locations of quadratic terms
291  int[] indices;
292  int which = 0;
293  functionDefinitions[which] = new int[0];
294  // linear terms:
295  for (int i = 0; i<dimension; ++i)
296  {
297  ++ which;
298  indices = new int[1];
299  indices[0] = i;
300  functionDefinitions[which] = indices;
301  }
302  // pure quadratic terms:
303  for (int i = 0; i<dimension; ++i)
304  {
305  ++ which;
306  indices = new int[2];
307  indices[0] = i;
308  indices[1] = i;
309  functionDefinitions[which] = indices;
310  quatraticTerms[i, i] = which;
311  }
312  // mixed terms:
313  for (int i = 0; i<dimension-1; ++i)
314  for (int j=i+1; j<dimension; ++j)
315  {
316  ++ which;
317  indices = new int[2];
318  indices[0] = i;
319  indices[1] = j;
320  functionDefinitions[which] = indices;
321  quatraticTerms[i, j] = quatraticTerms[i, j] = which;
322  }
323  if (which!=numFunc-1)
324  throw new InvalidOperationException("Quadratic basis functions: failed to initialize, wrong functions count.");
325  }

424  public override double Value(IVectorFunctionResults evaluationData, int which)
425  {
426  int[] funcDef = GetFunctionDefinition(which);
427  IVector param = evaluationData.Parameters;
428  if (funcDef.Length == 0)

431  return param[funcDef[0]];
432  else
433  {
434  if (funcDef[0]==funcDef[1])
435  return 0.5*param[funcDef[0]] * param[funcDef[1]]; // pure quadratic terms have factor 1/2
436  else
437  return param[funcDef[0]] * param[funcDef[1]];

448  public override double Derivative(IVectorFunctionResults evaluationData, int which,
449  int component)
450  {
451  IVector param = evaluationData.Parameters;
452  int[] funcDef = GetFunctionDefinition(which);
453  if (funcDef.Length == 0)
454  return 0;
455  else if (funcDef.Length == 1) // linear term
456  {
457  if (funcDef[0] == component)
458  return 1;
459  else
460  return 0;
461  } else
462  {
463  if (funcDef[0]==funcDef[1]) // pure quadratic term
464  {
465  if (component == funcDef[0])
466  return param[component]; // pure quadratic terms have factor 1/2
467  else
468  return 0;
469  } else // mixed quadratic term
470  {
471  if (component == funcDef[0])
472  return param[funcDef[1]];
473  else if (component == funcDef[1])
474  return param[funcDef[0]];
475  else
476  return 0;
477  }
478  }
