320  protected static void TestFunction(IFunc3d f,
321  double xmin, double xmax,
322  double ymin, double ymax,
323  double zmin, double zmax,
324  int numPoints)
325  {
326  Console.WriteLine("Values of the function at some points:");
327  for (int i = 0; i < numPoints; ++i)
328  {
329  double x = xmin + (double)i * (xmax - xmin) / ((double) numPoints - 1.0);
330  double y = ymin + (double)i * (ymax - ymin) / ((double)numPoints - 1.0);
331  double z = zmin + (double)i * (zmax - zmin) / ((double)numPoints - 1.0);
332  if (!f.ValueDefined)
333  Console.WriteLine("Function value is not defined.");
334  else
335  {
336  double value = f.Value(x, y, z);
337  Console.Write("x = " + x + ", y = " + y + ", z = " + z + "; f(x,y,z) = " + value);
338  double dx, dy, dz;
339  f.Gradient(x, y, z, out dx, out dy, out dz);
340  if (f.GradientDefined)
341  Console.Write(", grad f(x,z,y) = {" + dx + ", " + dy + ", " + dz + "}");
342  Console.WriteLine();
343  }
344  }
345  } // TestFunction(IFunc3d)
