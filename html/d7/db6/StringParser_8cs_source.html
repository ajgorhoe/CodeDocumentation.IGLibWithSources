221  public void AddEscapeSequence(char code, char character)
222  {
223  lock(_internalLock)
224  {
225  initEscapeSequences();
226  KeyValueSortable<char, char> item = new KeyValueSortable<char, char>(code, character);
227  int itemIndex = _escapeSequences.BinarySearch(item, KeyValueSortable<char, char>.CompareKeyValue);
228  if (itemIndex < 0)

230  // This escape sequence has not been added yet, do it now:

382  public int getChar(int from, int to, bool useEscape, ref char value)
383  {
384  lock(_internalLock)
385  {
386  int length = Length;
387  if (from < 0 || from > length || to < from)
