332  public static string RandomUnicodeString(int stringLength, bool excludeSurrogates = true, IRandomGenerator rndgen = null)
333  {
334  if (rndgen == null)
335  rndgen = RandomGenerator.Global;
336  StringBuilder plainText = new StringBuilder();
337  for (int j = 0; j < stringLength; ++j)
338  {
339  char next = (char) rndgen.Next(char.MinValue, char.MaxValue + 1);
340  if (next >= 0xD800 && next <= 0xDFFF)
341  {
342  // Surrogate character generated:
343  if (excludeSurrogates)
344  {
345  // We don't include surrogate characters. Keep generating until non-surrogate is obtained.
346  while(next >= 0xD800 && next <= 0xDFFF)
347  next = (char) rndgen.Next(char.MinValue, char.MaxValue + 1);
348  plainText.Append(next);
349  } else
350  {
351  // We include surrogate characters. These are only valid in proper sequences of two characters:
352  next = (char)rndgen.Next(0xD800, 0xDBFF + 1);
353  plainText.Append(next);
354  next = (char)rndgen.Next(0xDC00, 0xDFFF + 1);
355  plainText.Append(next);
356  }
357  } else
358  plainText.Append(next);
359  }
360  byte[] x = Encoding.Unicode.GetBytes(plainText.ToString());
361  string result = Encoding.Unicode.GetString(x);
362  return result;
363  }

399  public static char RandomChar(CharType typeflags, IRandomGenerator rndgen = null)
400  {
401  char ret = ' ';
402  if (rndgen == null)
403  rndgen = RandomGenerator.Global;
404  // typeflags defining whether certain types of characters are included:
405  bool
406  small = (typeflags & CharType.SmallLetter) != 0,
407  capital = (typeflags & CharType.CapitalLetter) != 0,
408  numeric = (typeflags & CharType.Numeric) != 0;
409  // Interval limits and weights for drawig lots:
410  double
411  smallint = 0.0, smallwt = 3.0,
412  capitalint = 0.0, capitalwt = 3.0,
413  numericint = 0.0, numericwt = 1.0,
414  max = 0.0;
415  // Define intervals for drawing lots:
416  if (small)
417  {
418  smallint = max + smallwt;
419  max = smallint;
420  }
421  if (capital)
422  {
423  capitalint = max + capitalwt;
424  max = capitalint;
425  }
426  if (numeric)
427  {
428  numericint = max + numericwt;
429  max = numericint;
430  }
431  if (max == 0.0)
432  throw new ArgumentException("Flags for random character type do not define any type.");
433  // Normalize intervals:
434  if (small)
435  smallint /= max;
436  if (capital)
437  capitalint /= max;
438  if (numeric)
439  numericint /= max;
440  // Select character type randomly:
441  double r = rndgen.NextDouble();
442  if (small && r <= smallint)
443  ret = RandomCharSmallLetter();
444  else if (capital && r <= capitalint)
445  ret = RandomCharCapitalLetter();
446  else if (numeric && r <= numericint)
447  ret = RandomCharNumeric();
448  else
449  throw new Exception("Could not generate a random character of the specified kind.");
450  return ret;
451  }
