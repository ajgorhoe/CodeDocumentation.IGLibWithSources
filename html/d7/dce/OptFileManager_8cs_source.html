83  {
84  lock (Lock)
85  {
86  if (String.IsNullOrEmpty(value))
87  throw new ArgumentException("Directory of the optimization/analysis file server is not specified (null or empty string).");
88  string dir = Path.GetFullPath(value);
89  if (!Directory.Exists(dir))
90  {
91  try
92  {

94  }
95  catch { }
96  if (!Directory.Exists(_directory))
97  throw new ArgumentException("The specified optimization/analysis file server directory does not exist. "
98  + Environment.NewLine + " Directory: " + value
99  + Environment.NewLine + " Full path: " + dir);
100  }

197  protected Mutex LockFileMutex
198  {
199  get {
200  if (_lockFileMutex == null)
201  {
202  lock (Lock)
203  {
204  if (_lockFileMutex == null)
205  {
206  bool createdNew;
207 #if NETFRAMEWORK
208  SecurityIdentifier sid = new SecurityIdentifier(WellKnownSidType.WorldSid, null);
209  MutexSecurity mutexsecurity = new MutexSecurity();
210  mutexsecurity.AddAccessRule(new MutexAccessRule(sid, MutexRights.FullControl, AccessControlType.Allow));
211  mutexsecurity.AddAccessRule(new MutexAccessRule(sid, MutexRights.ChangePermissions, AccessControlType.Deny));
212  mutexsecurity.AddAccessRule(new MutexAccessRule(sid, MutexRights.Delete, AccessControlType.Deny));
213  // _lockFileMutex = new Mutex(false, LockFileMutexName);
214  _lockFileMutex = new Mutex(false, LockFileMutexName,
215  out createdNew, mutexsecurity);
216 #else
217  _lockFileMutex = new Mutex(false, LockFileMutexName, out createdNew);
218 #endif
219  }
220  }
221  }
222  return _lockFileMutex;
223  }

675  public virtual bool IsAnInputReady()
676  {
677  bool ret = false;
678  bool lockAcquired = false;
679  try
680  {
681  lockAcquired = LockFileMutex.WaitOne(-1 /* wait indefinitely */);
682  if (File.Exists(MsgAnInputReadyPath))
683  ret = true;
684  else
685  ret = false;
686  }
687  catch (AbandonedMutexException ex)
688  {
689  if (ex.Mutex != null)
690  {
691  ex.Mutex.ReleaseMutex();
692  lockAcquired = false;
693  }
694  }
695  finally
696  {
697  if (lockAcquired)
698  LockFileMutex.ReleaseMutex();
699  }
700  return ret;
701  }
