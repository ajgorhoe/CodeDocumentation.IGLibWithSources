595  public virtual void CommandLine()
596  {
597  Console.Write(_commandLineHead);
598  string line, trimmed;
599  string inputCode = "";
600  string inputDefinitions = "";
601  bool evaluationMode = true;
602  bool multilineMode = false;
603  bool definitionMode = false;
604  bool stopCommandLine = false;
605  while (!stopCommandLine)
606  {
607  try
608  {
609  if (multilineMode)
610  Console.Write(_multilineMark);
611  else if (definitionMode)
612  Console.Write(_definitionsMark);
613  else
614  Console.Write(_inputMark);
615  line = Console.ReadLine();
616  trimmed = line.TrimEnd();
617  if (string.IsNullOrEmpty(trimmed))
618  {
619  // empty string inserted, just ignore
620  }
621  else if (trimmed == _quitCommand || trimmed == _quitCommand1 || trimmed == _quitCommand2)
622  {
623  stopCommandLine = true;
624  }
625  else if (trimmed == _helpCommand)
626  {
627  PrintHelpCommandLine();
628  }
629  else if (trimmed == _printDefinitionsCommand || trimmed == _printDefinitionsCommand1)
630  {
631  PrintDefinitions();
632  }
633  else if (trimmed == _printCodeCommand || trimmed == _printCodeCommand1)
634  {
635  PrintCompleteCode();
636  }
637  else if (trimmed == _saveCodeCommand || trimmed == _appendCodeCommand)
638  {
639  bool append = false;
640  if (trimmed == _appendCodeCommand)
641  append = true;
642  UserSaveCompleteCode(append);
643  }
644  else if (trimmed == _resetCommand)
645  {
646  Reset();
647  }
648  else if (trimmed == _evaluationCommand || trimmed == _evaluationCommand1
649  || trimmed == _multilineCommand || trimmed == _multilineCommand1
650  || trimmed == _definitionsCommand || trimmed == _definitionsCommand1)
651  {
652  // Perform all pending actions:
653  ExecuteUser(ref inputDefinitions, ref inputCode);
654  evaluationMode = multilineMode = definitionMode = false;
655  if (trimmed == _evaluationCommand || trimmed == _evaluationCommand1)
656  evaluationMode = true;
657  else if (trimmed == _multilineCommand || trimmed == _multilineCommand1)
658  {
659  Console.WriteLine("Multi-line mode. Input commands, {0} or {1} for evaluation.",
660  _evaluationCommand, _multilineCommand);
661  multilineMode = true;
662  }
663  else if (trimmed == _definitionsCommand || trimmed == _definitionsCommand1)
664  {
665  Console.WriteLine("Definitions mode. Input definitions, {0} or {1} to compile.",
666  _evaluationCommand, _definitionsCommand);
667  definitionMode = true;
668  }
669  }
670  else if (trimmed[0] == _commandCharacter)
671  {
672  // input line begins with command character, but command was not recognized,
673  // report this and ignore the input:
674  Console.WriteLine(" Unknown command: " + trimmed);
675  }
676  else if (trimmed[trimmed.Length - 1] == _multiLineCharacter
677  || multilineMode || definitionMode)
678  {
679  // Line ended with '\\' or in multiline mode, add input to concatenated lines read
680  // since the last complete input:
681  if (definitionMode)
682  inputDefinitions += Environment.NewLine + trimmed.TrimEnd(new char[] { _multiLineCharacter });
683  else // evaluation or multiline mode
684  inputCode += Environment.NewLine + trimmed.TrimEnd(new char[] { _multiLineCharacter });
685  }
686  else
687  {
688  // Last line of input (code or definitions), append and execute:
689  if (definitionMode)
690  inputDefinitions += Environment.NewLine + trimmed.TrimEnd(new char[] { _multiLineCharacter });
691  else // evaluation or multiline mode
692  inputCode += Environment.NewLine + trimmed.TrimEnd(new char[] { _multiLineCharacter });
693  ExecuteUser(ref inputDefinitions, ref inputCode);
694  }
695  }
696  catch (Exception ex)
697  {
698  Console.WriteLine();
699  Console.WriteLine("Error occurred in housekeeping code.");
700  if (!string.IsNullOrEmpty(ex.Message))
701  Console.WriteLine("Details: " + ex.Message);
702  Console.WriteLine();
703  }
704  }
705  Console.Write(_commandLineStopNote);
706  }
