553  public static void CalculateGridUnitFactors(int numElements, IRealFunction func,
554  double firstFunctionArgument, double lastFunctionArgument, ref List<double> factors)
555  {
556  if (numElements < 2)
557  {
558  numElements = 2;
559  throw new ArgumentException("Wrong number of elements in a grid (" + numElements
560  + "), should be at least 2.\n");
561  }
562  if (func == null)
563  throw new ArgumentNullException("Function for evaluation of grid factors is not defined.");
564  else if (!func.ValueDefined)
565  throw new ArgumentException("Function for evaluation of grid factors doesn't have its value defined. Function name. "
566  + func.Name + ".");
567  else if (firstFunctionArgument == lastFunctionArgument)
568  throw new ArgumentException("√ènterval on which grid distribution function is evaluated hay zero length.");
569  Util.ResizeList(ref factors, numElements, 0.0);
570  double firstValue = factors[0] = func.Value(firstFunctionArgument);
571  double lastValue = factors[numElements - 1] = func.Value(lastFunctionArgument);
572  double valueScalingFactor = 1.0/ ( (lastValue-firstValue) * (double) (numElements-1) );
573  double intervalLength = 1.0*(lastFunctionArgument-firstFunctionArgument)/(double) (numElements - 1);
574  for (int i=1; i<numElements-1; ++i)
575  factors[i] = func.Value(firstFunctionArgument + (double) i * intervalLength);
576  for (int i=0; i<numElements; ++i)
577  factors[i] = (factors[i]-firstFunctionArgument) * (double) i * valueScalingFactor;
578  } // CalculateTableUnitFactors
