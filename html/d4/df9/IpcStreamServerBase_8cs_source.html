147  protected virtual string ReadRequest()
148  {
149  lock (Lock)
150  {
151  try
152  {
153  RequestString = null;
154  string clientRequestString = null;
155  if (IsMultilineRequest)
156  {
157  if (OutputLevel >= 1)
158  {
159  Console.WriteLine(Environment.NewLine + "Waiting for a multiline request...");
160  }
161  StringBuilderInternal.Clear();
162  bool stop = false;
163  do
164  {
165  if (OutputLevel >= 2)
166  {
167  Console.WriteLine(" Waiting for next line of request...");
168  }
169  string line = InputStream.ReadLine();
170  if (OutputLevel >= 2)
171  {
172  Console.WriteLine("... line read: \"" + line + "\"");
173  }
174  if (line == MsgRequestEnd)
175  stop = true;
176  else
177  StringBuilderInternal.AppendLine(line);
178  } while (!stop);
179  clientRequestString = StringBuilderInternal.ToString();
180  clientRequestString = clientRequestString.TrimEnd('\n', '\r');
181  if (OutputLevel >= 1)
182  {
183  Console.WriteLine("... multiline request read: \"" + clientRequestString + "\"");
184  }
185 
186  }
187  else
188  {
189  if (OutputLevel >= 1)
190  {
191  Console.WriteLine(Environment.NewLine + "Waiting for a single lne request...");
192  }
193  clientRequestString = InputStream.ReadLine();
194  {
195  Console.WriteLine(Environment.NewLine + "... single line request read: \"" + clientRequestString + "\"");
196  }
197  }
198  RequestString = clientRequestString;
199  LastRequestString = clientRequestString;
200  IsResponseSent = false;
201  ResponseString = null;
202  IsError = false;
203  LastException = null;
204  LastErrorMessage = null;
205  }
206  catch (Exception ex)
207  {
208  LastRequestString = RequestString;
209  IsResponseSent = false;
210  ResponseString = null;
211  IsError = true;
212  LastException = ex;
213  LastErrorMessage = ex.Message;
214  }
215  return RequestString;
