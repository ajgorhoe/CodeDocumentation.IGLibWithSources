283  public static void MapCoordinatesReferencePlain(StructuredField2d<vec3> referenceField, StructuredField2d<vec3> targetField,
284  IFunc2d fx, IFunc2d fy, IFunc2d fz)
285  {
286  int dimx = referenceField.Dim1, dimy = referenceField.Dim2;
287  vec3 refCoords;
288  for (int j = 0; j < dimy; ++j)
289  for (int i = 0; i < dimx; ++i)

291  refCoords = referenceField[i, j];
292  targetField[i, j] = new vec3(fx.Value(refCoords.x, refCoords.y),
293  fy.Value(refCoords.x, refCoords.y), fz.Value(refCoords.x, refCoords.y));

391  public static void GenerateCoordinatesPlain(StructuredField2d<vec3> field,
392  double minxRef, double maxxRef,
393  double minyRef, double maxyRef)
394  {
395  double dimx = field.Dim1, dimy = field.Dim2;
396  double hx = 0, hy = 0, xRef, yRef;
397  if (dimx > 1)
398  hx = (maxxRef - minxRef) / (double)(dimx - 1);
399  if (dimy > 1)
400  hy = (maxyRef - minyRef) / (double)(dimy - 1);
401  for (int j = 0; j < dimy; ++j)
402  for (int i = 0; i < dimx; ++i)
403  {
404  xRef = minxRef + i * hx;
405  yRef = minyRef + j * hy;
406  field[i, j] = new vec3(xRef, yRef, 0.0);
407  }
408  }
