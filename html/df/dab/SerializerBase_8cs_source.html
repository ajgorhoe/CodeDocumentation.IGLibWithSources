126  public void Serialize<T>(T serializedObject, string filePath, bool append)
127  {
128  string directory = Path.GetDirectoryName(filePath);
129  if (!Directory.Exists(directory))
130  {
131  throw new ArgumentException("Directory containing the specified file does not exist. Path: " + filePath);
132  }
133  FileMode mode;
134  if (append)
135  mode = FileMode.Append;
136  else
137  mode = FileMode.Create;
138  using (FileStream fs = new FileStream(filePath, mode, FileAccess.Write))
139  {
140  Serialize<T>(serializedObject, fs);
141  }
142  }

328  public static void TestSerializationDto<TypeDto, Type, BaseType>(ISerializer serializer,
329  Type serializedObject, string filePath, bool firstStep, bool secondStep)
330  where BaseType : class
331  where Type : class, BaseType
332  where TypeDto : SerializationDtoBase<Type, BaseType>, new()
333  {
334  string path = Path.GetFullPath(filePath);
335  string filenamepure = Path.GetFileNameWithoutExtension(path);
336  string extension = Path.GetExtension(path);
337  string directory = Path.GetDirectoryName(path);
338 
339  if (!Directory.Exists(directory))
340  throw new ArgumentException("Directory of the specified file path does not exist. " + Environment.NewLine
341  + " Path: " + filePath);
342 
343  string pathOriginal = Path.Combine(directory,
344  filenamepure + extension);
345  string pathRestored = Path.Combine(directory,
346  filenamepure + "_restored" + extension);
347 
348  Console.WriteLine();
349  Console.WriteLine("Test of serialization/deserialization/serialization of an object to a file");
350  Console.WriteLine("Serialization is performed through a DTO.");
351  if (serializedObject!=null)
352  Console.WriteLine("Object type: " + serializedObject.GetType());
353  Console.WriteLine("Serializer: " + serializer.GetType());
354 
355  if (firstStep)
356  {
357  if (serializedObject == null)
358  {
359  Console.WriteLine();
360  Console.WriteLine("Object to be serialized is not specified, skipping serialization step.");
361  Console.WriteLine();
362  } else
363  {
364  Console.WriteLine("Copying object to DTO and writing serialized object to file: " + pathOriginal);
365  // Copy object to the corresponding DTO, serialize the DTO and store it to a file:
366  TypeDto dtoOriginal = new TypeDto();
367  dtoOriginal.CopyFrom(serializedObject);
368  serializer.Serialize<TypeDto>(dtoOriginal, pathOriginal);
369  }
370  }
371  if (secondStep)
372  {
373  // Desserialize the DTO corresponding to an object, copy it to a new object and back to a new DTO,
374  // and serialize it again to another file:
375  // First, deserialize object's state from file to a corresponding DTO:
376  Console.WriteLine("Restoring a corresponding DTO form the file (deserializing)...");
377  TypeDto restoredDto = serializer.DeserializeFile<TypeDto>(pathOriginal);
378  // Then copy DTO to object:
379  Console.WriteLine("Copying restored DTO to a new (restored) object:");
380  Type restoredObject = null;
381  restoredDto.CopyTo(ref restoredObject);
382  // Copying restored object to another DTO:
383  Console.WriteLine("Copying restored object to another DTO:");
384  TypeDto anotherDto = new TypeDto();
385  anotherDto.CopyFrom(restoredObject);
386  // Serialize another DTO to another file to enable comparison:
387  Console.WriteLine("Serializing another DTO to another file: ");
388  Console.WriteLine(" " + pathRestored);
389  serializer.Serialize<TypeDto>(anotherDto, pathRestored);
390  }
391  Console.WriteLine();
392  }
