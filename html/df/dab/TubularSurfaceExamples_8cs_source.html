126 
127  // Compute the parallel transport frame along the curve

129 
130  // Store points for the tubular surface

132 
133  for (int i = 0; i < tValues.Length; i++)

135  vec3 C_t = TubularSurface_01.Curve(tValues[i]); // Get curve position
136  vec3 N = frames[i].N;
137  vec3 B = frames[i].B;
138 
139  foreach (double theta in thetaValues)
140  {
141  // Compute point on the circular cross-section

143  tubePoints.Add(point);
