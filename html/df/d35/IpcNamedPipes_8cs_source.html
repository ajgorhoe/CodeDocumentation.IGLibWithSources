690  }

692  {
693  int milliSeconds = (int)(1000.0 * timeOutSeconds);
694  if (timeOutSeconds > 0 && milliSeconds <= 0)
695  milliSeconds = 1;
696  if (OutputLevel >= 1)
697  {
698  Console.WriteLine(Environment.NewLine + "Pipe client '" + PipeName + "': connecting with server, timeout: "

701  pipe.Connect(milliSeconds);
702 
703 
704  }
705  else
706  {
707  if (OutputLevel >= 1 && timeOutSeconds > 0 && !pipe.CanTimeout)
708  {
709  Console.WriteLine(Environment.NewLine + "WARNING: The client pipe does not support timeouts."
