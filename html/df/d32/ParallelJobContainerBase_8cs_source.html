461  public bool WaitJobCompletion(double timeoutInSeconds)
462  {
463  if (_outputLevel >= 1)
464  {
465  Console.WriteLine("Waiting job completion, timeout = " + timeoutInSeconds + " s..."
466  + Environment.NewLine + " job ID: " + Id
467  + Environment.NewLine + " disp. ID: " + DispatcherJobId
468  + Environment.NewLine + " cl. ID: " + ClientJobId);
469  }
470  if (IsJobCompleted)
471  return true;
472  else
473  {
474  bool completed = false, timeoutOccurred = false;
475  int sleepMs = SleepTimeMs, timeoutMs = (int)Math.Round(1e3 * timeoutInSeconds);
476  if (timeoutMs==0)
477  {
478  if (timeoutInSeconds>0) // prevent setting to 0 due to round-off
479  timeoutMs = 1;
480  }
481  int totalSleepTime = 0;
482  do
483  {
484  Thread.Sleep(sleepMs);
485  totalSleepTime += sleepMs;
486  completed = IsJobCompleted; // check whether the job has completed
487  if (!completed)
488  {
489  if (timeoutMs > 0 && totalSleepTime >= timeoutMs)
490  timeoutOccurred = true;
491  }
492  } while (!completed && !timeoutOccurred);
493  if (OutputLevel >= 1)
494  {
495  Console.WriteLine("Waiting job completion finished, job ID = " + Id + ", completed = " + completed);
496  }
497  return completed;
498  }
499  }

578  public static void TestPerformance(double from, double to, int numPoints, SimpleFunctionDelegate<double, double> evaluationFunction,
579  int numServers, int maxEnqueued, double delayTimeSeconds, double delayTimeRelativeError, int sleepTimeMs,
580  int clientOutputLevel)
581  {
582  Console.WriteLine();
583  Console.WriteLine("Testing parallel dispatcher by evaluating a real function of one variable...");
584  Console.WriteLine(" lower bound: " + from);
585  Console.WriteLine(" upper bound: " + to);
586  Console.WriteLine(" number of evaluation points: " + numPoints);
587  if (numPoints<=0)
588  throw new ArgumentException("Number of points must be greater than 1!");
589  double[] inputs = new double[numPoints];
590  double h = (to-from)/((double) numPoints-1.0);
591  for (int i=0; i<numPoints; ++i){
592  inputs[i] = from+i*h;
593  }
594  double [] results = ParallelJobContainerGen<double, double>.TestPerformance(inputs, evaluationFunction, numServers, maxEnqueued,
595  delayTimeSeconds, delayTimeRelativeError, sleepTimeMs, clientOutputLevel);
596  Console.WriteLine();
597  Console.WriteLine("Testing correctness of resuts calculated on parallel servers...");
598  Console.WriteLine(" number of inputs: " + numPoints);
599  Console.WriteLine(" number of results: " + results.Length);
600  int numWrong = 0;
601  int maxReported = 10;
602  int numReported = 0;
603  for (int i = 0; i < numPoints; ++i)
604  {
605  if (results[i] != evaluationFunction(inputs[i]))
606  {
607  ++numWrong;
608  ++numReported;
609  if (numReported <= maxReported)
610  {
611  Console.WriteLine("Result No. " + (i+1) + " is Wrong.");
612  }
613  }
614  }
615  if (numWrong == 0)
616  Console.WriteLine("... test finished, ALL results are CORRECT.");
617  else
