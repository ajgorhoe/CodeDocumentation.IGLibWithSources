156  public virtual bool CopyReferences
157  {
158  get { return _copyReferences; }
159  set
160  {
161  bool previous = _copyReferences;
162  _copyReferences = value;
163  if (previous == true && value == false)
164  {
165  // If we have copied references before, but now we copy values, we must
166  // make copies of all previous objects (we can do this simply by assignment
167  // because assignment will now perform deep copying rather than just reference copying):
168 
169  Copy(this);
170 
171  }
172  }
