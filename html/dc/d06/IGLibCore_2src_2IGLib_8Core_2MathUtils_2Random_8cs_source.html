336  public virtual double NextDouble(double maxValue)
337  {
338  double ret = maxValue * NextDouble();
339  if (ret == maxValue && maxValue!=0) // take care of exclusivity (mind rounding errors)
340  return NextDouble(maxValue);
341  return ret;
342  }

348  public virtual double NextDouble(double minValue, double maxValue)
349  {
350  double ret = minValue + NextDouble()*(maxValue-minValue);
351  if (ret == maxValue && maxValue != minValue) // take care of exclusivity (mind rounding errors)
352  return NextDouble(minValue, maxValue);
353  return ret;
354  }
