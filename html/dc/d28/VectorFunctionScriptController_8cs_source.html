190  public void SynchronizeData(bool syncParameterNames = false, bool syncFunctionNames = true)
191  {
192  if (Elements.Count >= NumValues)
193  Elements.RemoveRange(NumValues, Elements.Count - NumValues);
194  // Update parameter names and function names, if necessary:
195  bool updateParameterNames = false;
196  if (ParameterNames == null)
197  updateParameterNames = true;
198  else if (ParameterNames.Length != NumParameters)
199  updateParameterNames = true;
200  if (updateParameterNames)
201  {
202  string[] previousNames = ParameterNames;
203  string[] names = ScalarFunctionScriptController.GenerateParameterOrVariableNames(NumParameters, "x");
204  if (previousNames != null)
205  {
206  for (int i = 0; i < NumParameters; ++i)
207  if (!string.IsNullOrEmpty(previousNames[i]))
208  names[i] = previousNames[i];
209  }
210  this.ParameterNames = names;
211  }
212  bool updateFunctionNames = false;
213  if (FunctionNames == null)
214  updateFunctionNames = true;
215  else if (FunctionNames.Length != NumValues)
216  updateFunctionNames = true;
217  if (updateFunctionNames)
218  {
219  string[] previousNames = FunctionNames;
220  string[] names = ScalarFunctionScriptController.GenerateParameterOrVariableNames(NumParameters, "f");
221  if (previousNames != null)
222  {
223  for (int i = 0; i < NumValues; ++i)
224  if (!string.IsNullOrEmpty(previousNames[i]))
225  names[i] = previousNames[i];
226  }
227  this.FunctionNames = names;
228  }
229  // Synchronize state of scalar function controllers:
230  while (Elements.Count < NumValues)
231  {
232  AddScalarFunctionController(null);
233  }
234  for (int whichScalarFuction = 0; whichScalarFuction < NumValues; ++whichScalarFuction)
235  {
236  ScalarFunctionControllerType scalarController = this[whichScalarFuction];
237  if (scalarController == null)
238  {
239  this[whichScalarFuction] = scalarController = new ScalarFunctionControllerType();
240  scalarController.Dimension = this.NumParameters;
241  scalarController.ParameterNames = this.ParameterNames;
242  if (this.FunctionNames != null)
243  {
244  if (this.FunctionNames.Length > whichScalarFuction)
245  scalarController.Name = this.FunctionNames[whichScalarFuction];
246  }
247  } else
248  {
249  scalarController.Dimension = this.NumParameters;
250  if (syncParameterNames)
251  scalarController.ParameterNames = this.ParameterNames;
252  if (syncFunctionNames)
253  {
254  if (this.FunctionNames.Length > whichScalarFuction)
255  {
256  scalarController.Name = this.FunctionNames[whichScalarFuction];
257  }
258  }
259  }
260  }

517  public virtual InputOutputDataDefiniton GetDataDefinitionObject(bool setBoundsAndDefaults = false, IBoundingBox inputBounds = null, IBoundingBox outputBounds = null)
518  {
519  InputOutputDataDefiniton ret = new InputOutputDataDefiniton();
520  // string digitFormatString = new string('0', numNameDigits);
521  for (int i = 0; i < NumParameters; ++i)
522  {
523  InputElementDefinition inputElement = new InputElementDefinition(i, null, null);
524  inputElement.Name = inputElement.Title = ParameterNames[i];
525  inputElement.Description = "Function parameter " + i + "(" + ParameterNames[i] + ").";
526  if (setBoundsAndDefaults)
527  {
528  inputElement.MinimalValue = _defaultLowerBound;
529  inputElement.MaximalValue = _defaultUpperBound;
530  inputElement.DefaultValue = _defaultLowerBound;
531  inputElement.BoundsDefined = true;
532  inputElement.DefaultValueDefined = true;
533  if (inputBounds != null)
534  {
535  if (inputBounds.IsMinDefined(i))
536  inputElement.MinimalValue = inputBounds.GetMin(i);
537  else
538  inputElement.MinimalValue = double.MinValue;
539  if (inputBounds.IsMaxDefined(i))
540  inputElement.MaximalValue = inputBounds.GetMax(i);
541  else
542  inputElement.MaximalValue = double.MaxValue;
543  if (!(inputBounds.IsMinDefined(i) || inputBounds.IsMinDefined(i)))
544  inputElement.BoundsDefined = false;
545  if (inputElement.MinimalValue > inputElement.MaximalValue)
546  {
547  // swap lower and upper bound if necessary:
548  double min = inputElement.MinimalValue;
549  inputElement.MinimalValue = inputElement.MaximalValue;
550  inputElement.MaximalValue = min;
551  }
552  if (inputBounds.IsMinDefined(i) && inputBounds.IsMinDefined(i))
553  inputElement.DefaultValue = 0.5 * (inputElement.MinimalValue + inputElement.MaximalValue);
554  }
555  }
556  ret.AddInputElement(inputElement, true /* assign element index */);
557  }
558  for (int i = 0; i < NumValues; ++i)
559  {
560  OutputElementDefinition outputElement = new OutputElementDefinition(i, null, null);
561  outputElement.Name = outputElement.Title = FunctionNames[i];
562  outputElement.Description = "Function component " + i + "(" + FunctionNames[i] + ").";
563  if (setBoundsAndDefaults)
564  {
565  outputElement.MinimalValue = _defaultLowerBound;
566  outputElement.MaximalValue = _defaultUpperBound;
567  outputElement.BoundsDefined = true;
568  if (outputBounds != null)
569  {
570 
571  if (outputBounds.IsMinDefined(i))
572  outputElement.MinimalValue = outputBounds.GetMin(i);
573  else
574  outputElement.MinimalValue = double.MinValue;
575  if (outputBounds.IsMaxDefined(i))
576  outputElement.MaximalValue = outputBounds.GetMax(i);
577  else
578  outputElement.MaximalValue = double.MaxValue;
579  if (!(outputBounds.IsMinDefined(i) || outputBounds.IsMinDefined(i)))
580  outputElement.BoundsDefined = false;
581  if (outputElement.MinimalValue > outputElement.MaximalValue)
582  {
583  // swap bounds if necessary:
584  double min = outputElement.MinimalValue;
585  outputElement.MinimalValue = outputElement.MaximalValue;
586  outputElement.MaximalValue = min;
587  }
588 
589  }
590  }
591  ret.AddOutputElement(outputElement, true /* assign element index */);
592  }
593  return ret;
