100  public static int CsvColumnIndex(string[][] values, string fieldName, bool caseSensitive)
101  {
102  if (values != null) if (values.Length > 0)
103  {
104  int rowNum = 0;
105  bool foundNonEmpty = false;
106  while (rowNum < values.Length && !foundNonEmpty)
107  {
108  if (values[rowNum] != null)
109  if (values[rowNum].Length > 0)
110  {
111  if (values[rowNum].Length > 1)
112  foundNonEmpty = true;
113  else if (values[rowNum][0] != null)
114  if (values[rowNum][0].Length > 0)
115  foundNonEmpty = true;
116  }
117  if (!foundNonEmpty)
118  ++rowNum;
119  }
120  if (foundNonEmpty)
121  {
122  if (caseSensitive)
123  {
124  for (int columnNum = 0; columnNum < values[rowNum].Length; ++columnNum)
125  if (values[rowNum][columnNum] == fieldName)
126  return columnNum;
127  }
128  else
129  {
130  fieldName = fieldName.ToLower();
131  for (int columnNum = 0; columnNum < values[rowNum].Length; ++columnNum)
132  if (values[rowNum][columnNum].ToLower() == fieldName)
133  return columnNum;
134  }
135  }
136  }
137  return -1;
