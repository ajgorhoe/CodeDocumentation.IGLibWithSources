364  public void GetRelativeCoordinates(IVector coord, ref IVector relativeCoortinates)
365  {
366  if (coord == null)
367  throw new ArgumentException("The vector of coordinates is not specified.");
368  if (_min == null || _max == null)
369  throw new InvalidDataException("Vectors of minimal and maximal coordinates are not specified.");
370  int dim = _min.Length;
371  if (_max.Length != dim)
372  throw new InvalidDataException("Dimensions of minimal and maximal co-ordinate vectors do not match.");
373  if (coord.Length != dim)
374  throw new InvalidDataException("Dimension of vector of co-ordinates to be converted does not match dimensions of minimal and maximal coordinates ("
375  + dim + ").");
376  bool allocate = false;
377  if (relativeCoortinates == null)
378  allocate = true;
379  else if (relativeCoortinates.Length != dim)
380  allocate = true;
381  if (allocate)
382  relativeCoortinates = new Vector(dim);
383  for (int i = 0; i < dim; ++i)
384  {
385  double min = _min[i];
386  double max = _max[i];
387  double scalingLength = max - min;
388  if (scalingLength == 0)
389  scalingLength = 1;
390  relativeCoortinates[i] = (coord[i] - min) / scalingLength;
391  }
392  }

590  public virtual int Compare(PointContainerType pt1, PointContainerType pt2)
591  {
592  if (DistanceFunction==null)
593  throw new ArgumentException("Can not compare two points, distance function is not defined.");
594  if (ReferencePoint==null)
595  throw new ArgumentException("Can not compare two points, reference point is not defined.");
596  double d1 = DistanceFunction(pt1.Point, ReferencePoint.Point);
597  double d2 = DistanceFunction(pt2.Point, ReferencePoint.Point);
598  return d1<d2?-1:(d1>d2?1:0);
