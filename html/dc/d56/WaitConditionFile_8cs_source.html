386  public virtual void WaitEvents(int numEvents)
387  {
388  int caughtEvents = 0;
389  lock (TriggerLock)
390  {
391  try
392  {
393  IsWaiting = true;
394  CancelFlag = false;
395  if (PrintNotes)
396  Console.WriteLine("Blocking begins (waiting for " + numEvents + " matching file events).");
397  EnableRaisingEvents = true;
398  EventOccured = false;
399  while ( (numEvents > 0 && caughtEvents < numEvents)
400  && !CancelFlag)
401  {
402  Monitor.Wait(TriggerLock);
403  lock (TriggerLock)
404  {
405  if (EventOccured) // count events only if registered by this object's EventWatcher
406  {
407  ++caughtEvents;
408  EventOccured = false; // reset the flag
409  }
410  if (PrintNotes)
411  Console.WriteLine("Pulse caught in the waiting loop.");
412  }
413  }
414  if (PrintNotes)
415  Console.WriteLine("Specified number of events occurred, unblocked.");
416  }
417  catch (Exception) { }
418  finally
419  {
420  IsWaiting = false; if (CancelFlag) CancelFlag = false;
421  }
422  }
423  } // void WaitEvents(numEvents)
424 
426  public override void Wait()
427  {
428  lock (TriggerLock)
429  {
430  try
431  {
432  IsWaiting = true;
433  CancelFlag = false;
434  if (PrintNotes)
435  Console.WriteLine("Blocking begins (waiting for unblocking condition).");
436  EnableRaisingEvents = true;
437  while (!ConditionInternal() && !CancelFlag)
438  {
439  Monitor.Wait(TriggerLock);
440  lock (TriggerLock)
441  {
442  if (PrintNotes)
443  Console.WriteLine("Pulse catched in waiting loop.");
444  }
445  }
446  if (PrintNotes)
447  Console.WriteLine("Condition fulfilled, unblocked.");
448  }
449  catch (Exception) { }
450  finally {
451  IsWaiting = false; if (CancelFlag) CancelFlag = false;
452  }
453  }
454  } // void Wait()
