119  public override void ReferenceGradientPlain(IVector x, IVector gradient)
120  {
121  //if (x == null)
122  // throw new ArgumentException("Vector of parameters not specified when calculationg gradient of scalar function \"" + Name +"\" (nulll reference).");
123  int dim = x.Length;
124  if (gradient == null)
125  gradient = x.GetNew(dim);
126  else if (gradient.Length!=dim)
127  gradient = x.GetNew(dim);
128  double norm = x.Norm;
129  if (norm>Epsilon)
130  {
131  Vector.CopyPlain(x,gradient);

150  public override void ReferenceHessianPlain(IVector x, IMatrix hessian)
151  {
152  int dim = x.Length;
153  double normx = x.Norm;
154  double fDer = Function.Derivative(normx);
155  double fSecondDer = Function.SecondDerivative(normx);
156  if (normx < Epsilon)
157  {
158  // Close to zero, hessian matrix is diagonal, diagonal elements equal to second derivatives
159  // of 1-D function from which this scalar function is derived.
160  hessian.SetZero();
161  for (int i = 0; i < dim; ++i)
162  hessian[i, i] = fSecondDer;
163  } else
164  {
165  double normxSquare = normx * normx;
166  double normxCube = normxSquare * normx;
167  double element;
168  // Nondiagonal terms:
169  for (int i=1; i<dim; ++i)
170  for (int j = 0; j < i; ++j)
171  {
172  double prod_x_i_j = x[i]*x[j];
173  element = fSecondDer * prod_x_i_j / normxSquare
174  - fDer * prod_x_i_j / normxCube;
175  hessian[i,j] = element;
176  hessian[j, i] = element;
177  }
178  for (int i = 0; i < dim; ++i)
179  {
180  double x_i_square = x[i]; // square of component
181  x_i_square *= x_i_square;
182  hessian[i,i]=fSecondDer*x_i_square/normxSquare
183  + fDer*((1/normx)+(x_i_square/normxCube));
184  }
185  }
186  }
