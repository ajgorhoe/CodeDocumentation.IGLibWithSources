957  public virtual InputFieldDefinition GetInputFieldDefinition(int fieldId, string fieldName, string fieldDescription)
958  {
959  InputFieldDefinition def = null;
960  lock (Lock)
961  {
962  if (string.IsNullOrEmpty(fieldName))
963  {
964  if (fieldId <= 0)
965  throw new ArgumentException("Could not get an input field definition: neither field name nor Id is specified.");
966  foreach (InputFieldDefinition current in InputFieldDefinitions.Values)
967  {
968  if (current.Id == fieldId)
969  {
970  def = current;
971  // Update field description if specified by argument but not specified on already contained definition:
972  if (!string.IsNullOrEmpty(fieldDescription) && string.IsNullOrEmpty(current.Description))
973  current.Description = fieldDescription;
974  break;
975  }
976  }
977  if (def == null)
978  throw new ArgumentException("Could not get a field definition: field with ID " + fieldId + " is not defined.");
979  }
980  else
981  {
982  if (InputFieldDefinitions.ContainsKey(fieldName))
983  {
984  def = InputFieldDefinitions[fieldName];
985  if (fieldId > 0 && fieldId != def.Id)
986  throw new ArgumentException("Could not get an input field definition: Field ID " + fieldId
987  + " does not match that of the stored field definition (" + def.Id + ").");
988  // Update field description on the containing definition if currently not defined:
989  if (!string.IsNullOrEmpty(fieldDescription) && string.IsNullOrEmpty(def.Description))
990  def.Description = fieldDescription;
991  }
992  else
993  {
994  if (fieldId <= 0)
995  throw new ArgumentException("Can not get an input field definition: ID not specified and definition with name "
996  + def.Name + " does not yet exist.");
997  def = new InputFieldDefinition(fieldId, fieldName, fieldDescription);
998  // Field definition not yet contained in definitions, add it:
999  AddInputFieldDefinition(def);
1000  }
1001  }
1002  }
1003  if (def == null)
1004  throw new ArgumentException("Can not get input field definition, reason unknown. Field ID: "
1005  + def.Id + ", Name: " + def.Name + ".");
1006  return def;
1007  }
