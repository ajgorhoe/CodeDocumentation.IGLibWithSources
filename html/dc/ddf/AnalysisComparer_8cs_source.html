169  public int Compare(IAnalysisResults an1, IAnalysisResults an2)
170  {
171  if (an1 == null)
172  {
173  if (an2 != null)
174  return 1;
175  else
176  return 0;
177  }
178  else if (an2 == null)
179  {
180  return -1;
181  }
182  else
183  {
184  if (CompareNumViolated)
185  {
186  int nv1 = an1.GetNumViolatedConstraints();
187  int nv2 = an2.GetNumViolatedConstraints();
188  if (nv1 < nv2)
189  return -1;
190  else if (nv1 > nv2)
191  return 1;
192  }
193  if (CompareMaxResidual)
194  {
195  double mr1 = an1.GetMaximalResidual(EqualityConstraintTolerance);
196  double mr2 = an2.GetMaximalResidual(EqualityConstraintTolerance);
197  if (mr1 < mr2)
198  return -1;
199  else if (mr1 > mr2)
200  return 1;
201  }
202  if (CompareSumResiduals)
203  {
204  double sr1 = an1.GetMaximalResidual(EqualityConstraintTolerance);
205  double sr2 = an2.GetMaximalResidual(EqualityConstraintTolerance);
206  if (sr1 < sr2)
207  return -1;
208  else if (sr1 > sr2)
209  return 1;
210  }
211  if (CompareMaxPenalty)
212  {
213  double maxpen1 = an1.GetMaxPenaltyTerm(this.PenaltyEvaluator);
214  double maxpen2 = an1.GetMaxPenaltyTerm(this.PenaltyEvaluator);
215  if (maxpen1 < maxpen2)
216  return -1;
217  else if (maxpen1 > maxpen2)
218  return 1;
219  }
220  if (CompareSumPenalties)
221  {
222  double sumpen1 = an1.GetSumPenaltyTerms(this.PenaltyEvaluator);
223  double sumpen2 = an1.GetSumPenaltyTerms(this.PenaltyEvaluator);
224  if (sumpen1 < sumpen2)
225  return -1;
226  else if (sumpen1 > sumpen2)
227  return 1;
228  }
229  if (CompareObjectiveFunction)
230  {
231  double obj1 = an1.Objective;
232  double obj2 = an1.Objective;
233  if (obj1 < obj2)
234  return -1;
235  else if (obj1 > obj2)
236  return 1;
237  }
238  return 0;
239  }
240  }
