239  public void GetDistortedBounds(double DistortionFactor, double RandomFactor,
240  ref double distortedMinimalValue, ref double distrotedMaximalValue, IRandomGenerator randomGenerator)
241  {
242  if (DistortionFactor == 0.0)
243  DistortionFactor = 0.5;
244  if (RandomFactor < 0)
245  RandomFactor = -RandomFactor;
246  if (RandomFactor > 0.4)
247  RandomFactor = 0.4;
248  if (randomGenerator == null)
249  randomGenerator = RandomGenerator.Global;
250  double min = this.MinimalValue, max = this.MaximalValue;
251  if (this.BoundsDefined)
252  {
253  // First, multiply bounds by distortion factor.
254  if (DistortionFactor < 0)
255  {
256  // Swap bounds first if the factor is negative:
257  M.Swap(ref min, ref max);
258  }
259  min *= DistortionFactor;
260  max *= DistortionFactor;
261  // Then, define bounds for the distorted bounds:
262  double length = max - min;
263  double minLower = min - length * RandomFactor;
264  double minUpper = min + length * RandomFactor;
265  double maxLower = max - length * RandomFactor;
266  double maxUpper = max + length * RandomFactor;
267  // Correct if something changed its sign at this stage:
268  if (Math.Sign(minLower) != Math.Sign(min))
269  minLower = 0.5 * min;
270  if (Math.Sign(minUpper) != Math.Sign(min))
271  minUpper = 0.5 * min;
272  if (Math.Sign(maxLower) != Math.Sign(max))
273  maxLower = 0.5 * max;
274  if (Math.Sign(maxUpper) != Math.Sign(max))
275  maxUpper = 0.5 * max;
276  // Correct if the order of lower/upper bound limits is not correct:
277  if (minLower > minUpper)
278  M.Swap(ref minLower, ref minUpper);
279  if (maxLower > maxUpper)
280  M.Swap(ref maxLower, ref maxUpper);
281  if (minUpper > maxLower)
282  {
283  if (minUpper > maxUpper)
284  M.Swap(ref minUpper, ref maxUpper);
285  else if (minUpper > maxLower)
286  M.Swap(ref minUpper, ref maxLower);
287  if (minLower > maxLower)
288  M.Swap(ref minLower, ref maxLower);
289  if (minLower > minUpper)
290  M.Swap(ref minLower, ref minUpper);
291  if (maxLower > maxUpper)
292  M.Swap(ref maxLower, ref maxUpper);
293  }
294  // Determine the final values of distorted bounds:
295  if (minLower == minUpper)
296  min = minLower;
297  else
298  min = randomGenerator.NextDouble(minLower, minUpper);
299  if (maxLower == maxUpper)
300  max = maxLower;
301  else
302  max = randomGenerator.NextDouble(maxLower, maxUpper);
303  }
304  distortedMinimalValue = min;
305  distrotedMaximalValue = max;
306  }
