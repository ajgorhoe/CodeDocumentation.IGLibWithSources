66  public override void Evaluate(IVectorFunctionResults evaluationData)
67  {
68  if (evaluationData.NumFunctions <= 0)
69  evaluationData.NumFunctions = 2;
70  else if (evaluationData.NumFunctions != 2)
71  throw new InvalidDataException("Invalid number of vector function components: "
72  + evaluationData.NumFunctions + " instead of 2.");
73  if (evaluationData.NumParameters <= 0)
74  evaluationData.NumParameters = 2;
75  else if (evaluationData.NumParameters != 2)
76  throw new InvalidDataException("Invalid number of vector function parameters: "
77  + evaluationData.NumParameters + " instead of 2.");
78  IVector parameters = evaluationData.Parameters;
79  if (parameters == null)
80  throw new InvalidDataException("Vector of parameters not specified (null reference).");
81  else if (parameters.Length != 2)
82  throw new InvalidDataException("Invalid dimension of vector of parameters, "
83  + parameters.Length + " instead of 2.");
84  evaluationData.Calculated = false;
85  if (evaluationData.ReqValues)
86  {
87  evaluationData.AllocateValuesList();
88  evaluationData.Values[0] = F0.Value(parameters);
89  evaluationData.Values[1] = F1.Value(parameters);
90  evaluationData.CalculatedValues = true;
91  }
92  if (evaluationData.ReqGradients)
93  {
94  evaluationData.AllocateGradients();
95  IVector grad0 = evaluationData.Gradients[0];
96  F0.Gradient(parameters, ref grad0);
97  IVector grad1 = evaluationData.Gradients[1];
98  F1.Gradient(parameters, ref grad1);
99  evaluationData.Gradients[0] = grad0;
100  evaluationData.Gradients[1] = grad1;
101  evaluationData.CalculatedGradients = true;
102  }
103  if (evaluationData.ReqHessians)
104  {
105  evaluationData.AllocateHessians();
106  IMatrix hess0 = evaluationData.Hessians[0];
107  F0.Hessian(parameters, ref hess0);
108  IMatrix hess1 = evaluationData.Hessians[1];
109  F1.Hessian(parameters, ref hess1);
110  evaluationData.Hessians[0] = hess0;
111  evaluationData.Hessians[1] = hess1;
112  evaluationData.CalculatedGradients = true;
113  }
