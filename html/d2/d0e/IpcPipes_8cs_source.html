451  public static bool ExecuteSvnCommandWithFileInput(string command, string arguments, string filePath, out string result, out string errors)
452  {
453 
454 
455  bool retval = false;
456  string output = string.Empty;
457  string errorLines = string.Empty;
458  Process svnCommand = null;
459  var psi = new ProcessStartInfo(command);
460 
461 
462 
463  psi.RedirectStandardInput = true;
464  psi.RedirectStandardOutput = true;
465  psi.RedirectStandardError = true;
466  psi.WindowStyle = ProcessWindowStyle.Hidden;
467  psi.UseShellExecute = false;
468  psi.CreateNoWindow = true;
469 
470  try
471  {
472  Process.Start(psi);
473  psi.Arguments = arguments;
474  svnCommand = Process.Start(psi);
475 
476  var file = new FileInfo(filePath);
477  StreamReader reader = file.OpenText();
478  string fileContents = reader.ReadToEnd();
479  reader.Close();
480 
481  StreamWriter myWriter = svnCommand.StandardInput;
482  StreamReader myOutput = svnCommand.StandardOutput;
483  StreamReader myErrors = svnCommand.StandardError;
484 
485  myWriter.AutoFlush = true;
486  myWriter.Write(fileContents);
487  myWriter.Close();
488 
489  output = myOutput.ReadToEnd();
490  errorLines = myErrors.ReadToEnd();
491 
492  // Check for errors
493  if (errorLines.Trim().Length == 0)
494  {
495  retval = true;
496  }
497  }
498  catch (Exception ex)
499  {
500  string msg = ex.Message;
501  errorLines += Environment.NewLine + msg;
502  }
503  finally
504  {
505  if (svnCommand != null)
506  {
507  svnCommand.Close();
508  }
509  }
510 
511  result = output;
512  errors = errorLines;
513 
514  return retval;
515  }
