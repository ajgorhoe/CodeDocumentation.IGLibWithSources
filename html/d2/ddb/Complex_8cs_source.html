1061  }
1062  }
1063  // Check for nested open/closed brackets:
1064  if (s[0] == '(' || s[s.Length - 1] == ')')
1065  {
1066  if (s[0] == '(' && s[s.Length - 1] == ')')
1067  {
1068  throw new FormatException($"{nameof(Complex)}.{nameof(ParseRobust)}: The parsed string contains nested round brackets, which is currently not allowed in representation of a complex number.");
1069  }

1071  {
1072  throw new FormatException($"{nameof(Complex)}.{nameof(ParseRobust)}: The parsed string contains a nested open round bracket but no matching closed bracket.");
1073  }

1077  }
1078  }
1079  string sOriginalTrimmed = s;
1080  // After trimming round brackets, trim eventual leading and trailing whitespace again:
1081  s = s.Trim();
1082  // After trimming round brackets, check that there are no nested round brackets:
1083  // By now, we should have either just one floating point number (possibly with i) or two floating point numbers separated with comma,
1084  // and there should be no trailing spaces or round brackets:
1085 
1086  if (!s.Contains(','))
1087  {
1088  try
1089  {
1090  // s does not contain comma, meaning that the number either has only a real or only an imaginary part.
1091  // Try with the possibility of only the real part first:
1092  double real;
1093  bool success1 = double.TryParse(s, out real);

1096  return new Complex(real, 0);
1097  }
1098  // If it did not work, try with the posibility of only the imaginary part, which has 'i' in it (and possibly '*' on the left):
1099  if (s[s.Length - 1] == 'i')

1102  s = s.Substring(0, s.Length - 1);

1162  }

1166  // Try ti parse the imaginary part.
1167  if (identifiedSeparatorIndex == s.Length - 1)
1168  {
1169  throw new InvalidAsynchronousStateException($"{nameof(Complex)}.{nameof(ParseRobust)}: Malformed string, no number follows the last '{s[identifiedSeparatorIndex]}' character.");
