181  public virtual bool CopyReferences
182  {
183  get { return _copyReferences; }
184  set
185  {
186  bool previous = _copyReferences;
187  _copyReferences = value;
188  if (previous == true && value == false)
189  {
190  // If we have copied references before, but now we copy values, we must
191  // make copies of all previous objects (we can do this simply by assignment
192  // because assignment will now perform deep copying rather than just reference copying):
193  IVector tmpParameters = Parameters;
194  Parameters = null;
195  Parameters = tmpParameters;
196  List<double> tmpValues = Values;
197  Values = null;
198  Values = tmpValues;
199  List<IVector> tmpGradients = Gradients;
200  Gradients = null;
201  Gradients = tmpGradients;
202  List<IMatrix> tmpHessians = Hessians;
203  Hessians = null;
204  Hessians = tmpHessians;
205  }
