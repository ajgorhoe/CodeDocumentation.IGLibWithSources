916  double denom;
917  x2 -= x1; x3 -= x1; x4 -= x1;
918  y2 -= y1; y3 -= y1; y4 -= y1;
919  x1 = y1 = 0;

925  x3 * (x3*x3) * (x2 * y4 - x4 * y2) +
926  x3 * (x4 * (x4*x4) * y2 - x2 * (x2*x2) * y4);
