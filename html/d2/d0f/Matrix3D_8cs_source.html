875  public static int EigenSystem2d(mat3 a, ref mat3 eigenvec, ref vec3 eigenval)
876  {
877  double axx, axy, ayx, ayy, d;
878  axx = a.xx; axy = a.xy; ayx = a.yx; ayy = a.yy;
879  d = axx * axx + 4 * axy * ayx - 2 * axx * ayy + ayy * ayy;
880  if (d < 0)
881  {
882  /* The eigensystem has no real solutions: */
883  eigenvec.Zero();
884  eigenval.Zero();
885  return 0;
886  }
887  else
888  {
889  eigenvec.xz = eigenvec.yz = eigenvec.zz = eigenvec.zx = eigenvec.zy =
890  eigenval.z = 0;
891  d = Math.Sqrt(d);
892  if (d > 1e-10 * Math.Abs(axx + ayy))
893  {
894  /* We have two real eigenvalues: */
895  if (Math.Abs(ayx) / (Math.Abs(axx) + Math.Abs(ayy) + Math.Abs(d)) > 1e-20)
896  {
897  eigenval.x = (axx + ayy - d) / 2;
898  eigenvec.xx = -(-axx + ayy + d) / (2 * ayx);
899  eigenvec.xy = 1;
900  eigenval.y = (axx + ayy + d) / 2;
901  eigenvec.yx = -(-axx + ayy - d) / (2 * ayx);
902  eigenvec.yy = 1;
903  }
904  else
905  {
906  /* When ayx==0 we must adopt another formula: */
907  eigenval.x = axx;
908  eigenvec.xx = 1;
909  eigenvec.xy = 0;
910  eigenval.y = ayy;
911  if (Math.Abs(axx - ayy) > 1e-20)
912  {
913  eigenvec.yx = -(axy / (axx - ayy));
914  eigenvec.yy = 1;
915  }
916  else
917  {
918  /* To ni striktno, velja pa za simetricne matrike (zaradi ortog. last.
919  vektorjev): */
920  eigenvec.yx = 0;
921  eigenvec.yy = 1;
922  }
923  }
924  // Normalize eigenvectors:
925  eigenvec.rowx = eigenvec.rowx.Normalized2();
926  eigenvec.rowy = eigenvec.rowy.Normalized2();
927  return 2;
