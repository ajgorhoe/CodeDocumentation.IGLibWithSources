352  public string CreateResponseOrRequestString(string originalResponseOrRequestString, bool multiLine)
353  {
354  if (originalResponseOrRequestString == null)
355  return null;
356  else
357  {
358  string ret = null;
359  if (!originalResponseOrRequestString.StartsWith(MessagePrefix))
360  ret = originalResponseOrRequestString;
361  else
362  ret = MessagePrefix + MessageFalseSeparator + originalResponseOrRequestString.Substring(MessagePrefix.Length);
363  if (!multiLine)
364  ret = ret.Replace("\n", "\\n").Replace("\r","\\r");
365  return ret;
366  }

376  public void InterpretRequestOrResponseLine(ref string responseOrRequestString, out bool isMessage,
377  out string messageOrCommandName, out string [] messageArguments)
378  {
379  isMessage = false;
380  messageOrCommandName = null;
381  messageArguments = null;
382  if (responseOrRequestString != null)
383  {
384  if (responseOrRequestString.StartsWith(MessagePrefix))
385  {
386  if (responseOrRequestString.StartsWith(MessagePrefixWithSeparator))
387  {
388  // String is actually a message:
389  isMessage = true;
390  responseOrRequestString = responseOrRequestString.Substring(MessagePrefixWithSeparator.Length);
391  string[] parts = responseOrRequestString.Split(new char[] {' ', '\t', '\r', '\n'}, StringSplitOptions.RemoveEmptyEntries);
392  messageOrCommandName = parts[0];
393  int numParts = parts.Length;
394  if (numParts > 1)
395  {
396  messageArguments = new string[numParts - 1];
397  for (int i = 1; i < numParts; ++i)
398  messageArguments[i-1] = parts[i];
399  }
400  } else
401  {
402  // String begins with message prfix, but it is not a message (because a separator is not following):
403  // Actual request/response string is obtained by excluding the inserted additional character after the MessagePrefix:
404  responseOrRequestString = MessagePrefix + responseOrRequestString.Substring(MessagePrefix.Length + 1);
405  }
406  }
407  }

416  public void WriteMessage(StreamWriter outputStream, string message, string[] args)
417  {
418  if (message.Contains('\n'))
419  throw new InvalidDataException("Invalid message name: contains newlines.");
420  outputStream.Write(MessagePrefix);
421  outputStream.Write(MessageSeparator);
422  outputStream.Write(message);
423  if (args != null)
424  {
425  int numArgs = args.Length;
426  for (int i = 0; i < numArgs; ++i)
427  {
428  string argStr = args[i];
429  if (string.IsNullOrEmpty(argStr))
430  outputStream.Write(" \"\"");
431  else
432  {
433  if (argStr.Contains('\n'))
434  throw new InvalidDataException("Invalid message argument No. " + i + ": contains newlines.");
435  outputStream.Write(" ");
436  outputStream.Write(argStr);
437  }
438  }
439  }
440  outputStream.WriteLine();
441  }
