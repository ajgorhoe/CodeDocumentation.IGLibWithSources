220  public override bool RepairInputParameters(IVector parameters)
221  {
222  SimKosecFileManagerConvBase sim = this ;
223  int dim = 2;
224  bool ret = false;
225  if (parameters == null)
226  throw new ArgumentException("Vector of parameters is null.");
227  if (parameters.Length != dim)
228  throw new ArgumentException("Wrong dimension of vector of parameters: " + parameters.Length
229  + " instead of " + dim + ".");
230  double p0 = parameters[0];
231  p0 = sim.MinX + Math.Round((p0 - sim.MinX) / sim.DX) * sim.DX;
232  if (p0 != parameters[0])
233  {
234  ret = true;
235  parameters[0] = p0;
236  }
237  double p1 = parameters[1];
238  p1 = sim.MinY + Math.Round((p1 - sim.MinY) / sim.DY) * sim.DY;
239  if (p1 != parameters[1])
240  {
241  ret = true;
242  parameters[1] = p1;
243  }
244  return ret;
245  }

272  protected override void UpdateInputParametersManual(IVector inputParameters)
273  {

275  throw new ArgumentException("Vector of input parameters not specified (null reference).");
276  else if (inputParameters.Length != NumInputParameters)
277  throw new ArgumentException("Dimension of input parameters is different than " + NumInputParameters + ".");

279  double x2_1 = inputParameters[1];
280  double x1_2 = x1_1 + ObstacleLength;

288  if (x2_2 > MaxX)
289  throw new ArgumentException("x2_2 < " + MaxX + ".");
290  SetInputFieldValue("Obstacle1:x0", x1_1.ToString());
