171  public override void Evaluate(IVector parameters, ref bool calculateValues, ref List<double> values,
172  ref bool calculateGradients, ref List<IVector> gradients,
173  ref bool calculateHessians, ref List<IMatrix> hessians)
174  {
175  lock (Lock)
176  {
177  IVectorFunctionResults results = ResultsStore.GetEligible();
178  if (results == null)
179  throw new InvalidOperationException("Vector function: could not get auxiliary results object from the store.");
180  results.Parameters = parameters;
181  results.ReqValues = calculateValues;
182  results.ReqGradients = calculateGradients;
183  results.ReqHessians = calculateHessians;
184  // Actual evaluation:
185  Evaluate(results);
186  if (calculateValues && !results.CalculatedValues)
187  {
188  string message = "Vector function values could not be evaluated.";
189  if (results.ErrorCode < 0)
190  message += Environment.NewLine + " Error code: " + results.ErrorCode;
191  if (!String.IsNullOrEmpty(results.ErrorString))
192  message += Environment.NewLine + " Error string: " + results.ErrorString;
193  throw new InvalidOperationException(message);
194  }
195  if (calculateGradients && !results.CalculatedGradients)
196  {
197  string message = "Vector function gradients could not be evaluated.";
198  if (results.ErrorCode < 0)
199  message += Environment.NewLine + " Error code: " + results.ErrorCode;
200  if (!String.IsNullOrEmpty(results.ErrorString))
201  message += Environment.NewLine + " Error string: " + results.ErrorString;
202  throw new InvalidOperationException(message);
203  }
204  if (calculateHessians && !results.CalculatedHessians)
205  {
206  string message = "Vector function hessians could not be evaluated.";
207  if (results.ErrorCode < 0)
208  message += Environment.NewLine + " Error code: " + results.ErrorCode;
209  if (!String.IsNullOrEmpty(results.ErrorString))
210  message += Environment.NewLine + " Error string: " + results.ErrorString;
211  throw new InvalidOperationException(message);
212  }
213  int numFunc = results.NumFunctions;
214  int numParam = results.NumParameters;
215  if (calculateValues && results.CalculatedValues)
216  {
217  Util.ResizeList(ref values, numFunc, 0.0, true);
218  for (int i = 0; i < numFunc; ++i)
219  values[i] = results.GetValue(i);
220  }
221  if (calculateGradients && results.CalculatedGradients)
222  {
223  Util.ResizeList(ref gradients, numFunc, null, true);
224  for (int which = 0; which < numFunc; ++which)
225  {
226  IVector gradient = gradients[which];
227  if (gradient == null)
228  gradient = gradients[which] = new Vector(numParam);
229  else if (gradient.Length != numParam)
230  gradient = gradients[which] = new Vector(numParam);
231  for (int i = 0; i < numParam; ++i)
232  gradient[i] = results.GetGradient(which, i);
233  }
234  }
235  ResultsStore.StoreEligible(results);
236  if (calculateHessians && results.CalculatedHessians)
237  {
238  Util.ResizeList(ref hessians, numFunc, null, true);
239  for (int which = 0; which < numFunc; ++which)
240  {
241  IMatrix hessian = hessians[which];
242  if (hessian == null)
243  hessian = hessians[which] = new Matrix(numParam, numParam);
244  else if (hessian.RowCount != numParam
245  || hessian.ColumnCount != numParam)
246  hessian = hessians[which] = new Matrix(numParam, numParam);
247  for (int i = 0; i < numParam; ++i)
248  for (int j = 0; j < numParam; ++j)
249  hessian[i, j] = results.GetHessian(which, i, j);
250  }
251  }
252  ResultsStore.StoreEligible(results);
253  }
254  } // Evaluate(IVector, ref bool, ref List<double>, ref bool, ref List<IVector>, ref bool, ref List<IMatrix>)

294  public override void Evaluate(IVectorFunctionResults evaluationData)
295  {
296  if (evaluationData==null)
297  throw new InvalidOperationException("Vector function: evaluation data is not specified.");
298  if (evaluationData.Parameters == null)
299  throw new InvalidOperationException("Vector function: parameters are not specified.");
300  int numPar = this.NumValues;
301  int numFunc = this.NumValues;
302  // Set basic propeties of vector function on evaluation data:
303  evaluationData.NumParameters = numPar;
304  evaluationData.NumFunctions = numFunc;
305  // Extract calculatio flags from evaluation data:
306  bool calculateValues = evaluationData.ReqValues;
307  bool calculateGradients = evaluationData.ReqGradients;
308  bool calculateHessians = evaluationData.ReqHessians;
309  // Get references to objects holding individual results on evaluation data:
310  List<double> values = evaluationData.Values;
311  List<IVector> gradients = evaluationData.Gradients;
312  List<IMatrix> hessians = evaluationData.Hessians;
313  // Perform evaluation by another function:
314  Evaluate(evaluationData.Parameters, ref calculateValues, ref values,
315  ref calculateGradients, ref gradients,
316  ref calculateHessians, ref hessians);
317  // Store flags that indicate what has been calculated:
318  evaluationData.CalculatedValues = calculateValues;
319  evaluationData.CalculatedGradients = calculateGradients;
320  evaluationData.CalculatedHessians = calculateHessians;
321  // Set result references on evaluation data to possibly modified objects
322  // (because evaluation function could have changed references, too):
323  evaluationData.SetValuesReference(values);
324  evaluationData.SetGradientsReference(gradients);
325  evaluationData.SetHessiansReference(hessians);
326  int errorCode = 0;
327  string errorString = null;
328  if (evaluationData.ReqValues && !calculateValues)
329  {
330  errorCode = -1;
331  errorString = errorString + "Vector functions could not be evaluated.";
332  }
333  if (evaluationData.ReqGradients && !calculateGradients)
334  {
335  if (errorCode>=0)
336  errorCode = -2;
337  errorString = errorString + "Vector function gradients could not be evaluated.";
338  }
339  if (evaluationData.ReqHessians && !calculateHessians)
340  {
341  if (errorCode >= 0)
342  errorCode = -1;
343  errorString = errorString + "Vector function hessians could not be evaluated.";
344  }

402  public override void Evaluate(IVector parameters, ref bool calculateValues, ref List<double> values,
403  ref bool calculateGradients, ref List<IVector> gradients,
404  ref bool calculateHessians, ref List<IMatrix> hessians)
405  {
406  int numFunc = NumValues;
407  int numParam = parameters.Length;
408  if (numParam <= 0)
409  throw new InvalidOperationException("Vector function in inconsystent state: number of parameters is 0.");
410  if (numFunc <= 0)
411  throw new InvalidOperationException("Vector function in inconsystent state: number of functions is 0.");
412  IVectorFunctionResults results = null;
413  results = ResultsStore.TryGet();
414  if (results == null)
415  results = new VectorFunctionResults();
416  results.NumFunctions = numFunc;
417  results.SetParametersReference(parameters);
418  // Calculaton:
419  if (calculateValues)
420  {
421  Util.ResizeList(ref values, numFunc, 0.0, true);
422  for (int which = 0; which < numFunc; ++which)
423  values[which] = Value(results, which);
424  }
425  if (calculateGradients)
426  {
427  Util.ResizeList(ref gradients, numFunc, null, true);
428  for (int which = 0; which < numFunc; ++which)
429  {
430  IVector gradient = gradients[which];
431  if (gradient == null)
432  gradient = gradients[which] = new Vector(numParam);
433  else if (gradient.Length != numParam)
434  gradient = gradients[which] = new Vector(numParam);
435  for (int i = 0; i < numParam; ++i)
436  gradient[i] = Derivative(results, which, i);
437  }
438  }
439  if (calculateHessians)
440  {
441  Util.ResizeList(ref hessians, numFunc, null, true);
442  for (int which = 0; which < numFunc; ++which)
443  {
444  IMatrix hessian = hessians[which];
445  if (hessian == null)
446  hessian = hessians[which] = new Matrix(numParam, numParam);
447  else if (hessian.RowCount != numParam
448  || hessian.ColumnCount != numParam)
449  hessian = hessians[which] = new Matrix(numParam, numParam);
450  for (int i = 0; i < numParam; ++i)
451  for (int j = 0; j < numParam; ++j)
452  {
453  hessian[i, j] = SecondDerivative(results, which, i, j);
454  }
455  }
456  }
457  lock (Lock)
458  {
459  // Store results back to auxiliary variable:
460  results.SetParametersReference(null);
