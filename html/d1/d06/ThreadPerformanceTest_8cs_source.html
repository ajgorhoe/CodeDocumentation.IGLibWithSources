69  public static void StandardCycle()
70  {
71  double a, b, c, d;
72  int i;
73  for (i=0; i < NumInner; ++i)
74  {
75  a = Math.Exp(0.1);
76  b = Math.Sin(a) + Math.Cos(a) + 0.1 * Math.Sqrt(a * a);
77  c = Math.Sqrt(a * b) + Math.Cos(a + b);
78  d = (a + b) * (b + c) * (c - a) * Math.Sqrt(a * b * c);
79  }

144  public void TestPerformance(double requestedDuration, out ThreadPerformanceData data)
145  {
146  int numCycles = 0;
147  int numIt = 0;
148  double cyclesLeft = 0; // estimated number of cycles yet to be performed in order to reach the requested measurement time
149  double prevCycles = 0; // previous number of cycles
150  double timeLeft = requestedDuration;
151  double timePortion = 0.0;
152  double CPS = RefCyclesPerSecond; // current cycles per second
153  double cyclesFactor = 0.05;
154  if (Last.NumCycles > 0)
155  {
156  // Measurement has already been performed, so we have an estimation of cycles per second:
157  CPS = Last.CyclesPerSecond;
158  cyclesFactor = 0.2; // still we consider first estimation very rough:
159  } else
160  {
161  // We don't have any prior estimation of performcne, so we take the reference number:
162  CPS = RefCyclesPerSecond;
163  cyclesFactor = 0.05; // since estimation is very bad
164  }
165  cyclesLeft = CPS * timeLeft;
166  cyclesLeft *= cyclesFactor;
167  if (cyclesLeft < MinCycles)
168  cyclesLeft = MinCycles;
169 
170  StopWatch1 t = new StopWatch1();
171  while (timePortion < 0.98)
172  {
173  ++numIt;
174  prevCycles = cyclesLeft;
175  //Console.WriteLine("Iteration " + numIt + ": ");
176  //Console.WriteLine(" timePortion: " + timePortion + " timeLeft: " + timeLeft);
177  //Console.WriteLine(" CPS: " + CPS + " cyclesLeft: " + cyclesLeft);
178 
179  t.Start();
180  for (int i = 0; i <= cyclesLeft; ++i)
181  {
182  StandardCycle();
183  ++numCycles;
184  }
185  t.Stop();
186  if (t.TotalTime == 0)
187  CPS = RefCyclesPerSecond;
188  else
189  CPS = numCycles / t.TotalTime; // improve estimation of cycles per second
190  timeLeft = requestedDuration - t.TotalTime;
191  timePortion = 1.0 - (timeLeft / requestedDuration); // portion of time spent w.r. prescribed duration
192  cyclesLeft = CPS * timeLeft; // estimation of number of cycles needed to reach the requested duration
193  cyclesFactor = 1.0;
194  if (timePortion < 0.05)
195  cyclesFactor = 0.1; // since we have a very rough estimation
196  else if (timePortion < 0.1)
197  cyclesFactor = 0.2;
198  else if (timePortion < 0.5)
199  cyclesFactor = 0.6;
200  else if (timePortion < 0.8)
201  cyclesFactor = 0.95;
202  cyclesLeft *= cyclesFactor;
203  if (cyclesLeft < MinCycles)
204  cyclesLeft = MinCycles;
205  if (t.TotalTime <= 0.005)
206  cyclesLeft = prevCycles * 1.5; // safety: time interval too small to measure
207  }
208  Last.Time = DateTime.Now;
209  Last.NumCycles = numCycles;
210  Last.CyclesPerSecond = (double)numCycles / t.TotalTime;
211  Last.PerformanceRatio = t.TotalCpuTime / t.Time;
212  Last.TotalTime = t.TotalTime;
213  Last.NumIterations = numIt;
214  StoreLast();
215  data = Last;
216  }
