267  {

270  InputType input = default(InputType);
271  ResultType result = default(ResultType);

277  if (IsTestMode)

279  int num = NumTestDelays;
280  int delayMs = TestDelaySingleMs;
281  if (num>0 && delayMs>0)
282  {
283  for (int i = 0; i<num; ++i)
284  {
285  if (OutputLevel>=2)
286  {
287  if (num>1)
288  {
289  Console.WriteLine("Job No. " + Id + ", sleeping for " + ((double)delayMs/1.0e3) + "s, "
290  + (i+1) + "/" + num + "...");
291  } else
292  {
293  Console.WriteLine("Job No. " + Id + ", sleeping for " + ((double)delayMs/1.0e3) + "s...");
294  }
295  }
296  Thread.Sleep(delayMs);
297  }
298  }
299  }
300  result = CalculateResultsPlain(input);
301  lock(Lock)
302  {
303  Result = result;
304  State = ParallelJobState.ResultsReady;
305  }
306  if (OutputLevel >= 1)

308  Console.WriteLine("... job (by container's method) finished, container No. " + this.Id + ".");

466  };
467  // Set callback for job finished event, assign anonymous function:
468  job.OnFinishedGeneric = delegate(ParallelJobContainerGen<InputType, ResultType> jobContaier)
469  {
470  // Find out which result is contained on the job container - through client index:
471  int which = jobContaier.ClientJobId;
472  if (clientOutputLevel>=2)
473  Console.WriteLine(">> Finished event triggered, client ID = " + jobContaier.ClientJobId);
474  // Resize result list, if necessary:
475  while (which >= results.Count)
476  {
477  results.Add(default(ResultType));
478  }
479  // Store the result:
480  results[which] = jobContaier.Result;
481  };
482  if (clientOutputLevel >= 2)
483  Console.WriteLine(Environment.NewLine
484  + " >> max. enqueued jobs: " + maxEnqueued + ", num. enqueued: " + dispatcher.NumEnqueuedJobs
485  + ", num. idle servers: " + dispatcher.NumIdleJobServers);

487  {
488  Thread.Sleep(10);
489  }
490  if (clientOutputLevel >= 2)
491  Console.WriteLine(""
492  + " >> max. enqueued jobs: " + maxEnqueued + ", num. enqueued: " + dispatcher.NumEnqueuedJobs
493  + ", num. idle servers: " + dispatcher.NumIdleJobServers);
494  int jobNumber = job.Id;
495  if (clientOutputLevel >= 1)
496  Console.WriteLine(">> Sendinig job ID = {0} ... ", jobNumber);
497  dispatcher.SendJob(job);
498  if (clientOutputLevel >= 3)

500  Console.WriteLine(" >> ...job ID = {0} sent. ", jobNumber);
501  Console.WriteLine(">> jobs dispatched: "

503  + Environment.NewLine + " total started: " + dispatcher.NumStartedJobs
504  + Environment.NewLine + " finished: " + dispatcher.NumFinishedJobs
505  + Environment.NewLine + " aborted: " + dispatcher.NumAbortedJobs
506  + Environment.NewLine + " uncompleted: " + dispatcher.NumUncompletedJobs
507  + Environment.NewLine + " enqueued: " + dispatcher.NumEnqueuedJobs
508  + Environment.NewLine + " executing: " + dispatcher.NumExecutingJobs);

510  }

541  }
