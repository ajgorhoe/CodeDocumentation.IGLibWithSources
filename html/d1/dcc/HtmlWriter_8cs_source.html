587  public void AppendAttributes(string[] attributeNameValuePairs)
588  {
589  if (IsAttributesDefined(attributeNameValuePairs))
590  {
591  if (attributeNameValuePairs.Length % 2 != 0)
592  throw new ArgumentException("Number of provided fields and values is not even.");
593  AppendPlainText(" ");
594  // AppendPlainText(GetIndent() + "<meta ");
595  int which = 0;
596  while (which < attributeNameValuePairs.Length)
597  {
598  string fieldName = attributeNameValuePairs[which];
599  ++which;
600  if (string.IsNullOrEmpty(fieldName))
601  throw new ArgumentException("Attributes: name of the field No. " + (which % 2) + " is not specified. (null or empty string).");
602  string fieldValue = attributeNameValuePairs[which];
603  AppendPlainText(fieldName + "=\"" + fieldValue + "\"");
604  if (which < attributeNameValuePairs.Length - 1)
605  AppendPlainText(" ");
606  ++which;
607  }

769  public void AddImage(string imageLink, string altText, string captionText, int width, int height,
770  int originalWidth, int originalHeight, bool constrainRatio)
771  {
772  if (width <= 0 || height <= 0)
773  {
774  if (constrainRatio)
775  {
776  if (width <= 0 && height <= 0)
777  throw new ArgumentException("Both image width and height are less or equal to 0.");
778  } else
779  {
780  throw new ArgumentException("Image width or heighth is leee or equal than 0 while ratio is not constrained.");
781  }
782  }
783  if (constrainRatio)
784  {
785  if (originalWidth <= 0 || originalHeight <= 0)
786  throw new ArgumentException("Original image width or height not specified while ratio between width and height is constrained.");
787  double originalRatio = (double) originalWidth/(double) originalHeight;
788  double ratio = 1.0;
789  if (width>0 && height>0)
790  ratio = (double)width / (double)height;
791  if (width<=0)
792  // Width not specified, calculate it from height:
793  width = (int)Math.Round((double) height * ratio);
794  else if (height<=0)
795  // Height not specifid, calculate it form width:
796  height = (int)Math.Round((double)width / ratio);
797  else if (ratio > originalRatio)
798  // Specified width is too large, reduce it in order to correspond to hwight:
799  width = (int)Math.Round((double) height * ratio);
800  else if (ratio < originalRatio)
801  // Specified width is too large, reduce it in order to correspond to hwight:
802  height = (int)Math.Round((double)width / ratio);
803  }
804  AddImage(imageLink, altText, captionText, width, height, true /* defineSize */ );
805  }
