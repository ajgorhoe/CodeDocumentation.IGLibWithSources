172  protected virtual void Report_Console(ReportType messagetype, string location, string message, Exception ex)
173  {
174  string locationstring = "", messagestring = "";
175  if (ReportLocationDlgConsole != null)
176  locationstring = ReportLocationDlgConsole(this, messagetype, location, ex);
177  else if (ReportLocationDlg != null)
178  locationstring = ReportLocationDlg(this, messagetype, location, ex);
179  else
180  {
181  // No delegate for assembling location string:
182  if (!string.IsNullOrEmpty(location))
183  locationstring += location;
184  }
185  if (ReportMessageDlgConsole != null)
186  messagestring = ReportMessageDlgConsole(this, messagetype, message, ex);
187  else if (ReportMessageDlg != null)
188  messagestring = ReportMessageDlg(this, messagetype, message, ex);
189  else
190  {
191  // No delegate for assembling message string:
192  if (!string.IsNullOrEmpty(messagestring))
193  {
194  messagestring += message;
195  if (ex != null) if (!string.IsNullOrEmpty(ex.Message))
196  messagestring += " Details: ";
197  }
198  if (ex != null) if (!string.IsNullOrEmpty(ex.Message))
199  messagestring += ex.Message;
200  }
201  if (ReportDlgConsole != null)
202  ReportDlgConsole(this, messagetype, locationstring, messagestring);
203  else
204  throw new Exception("Console form reporting delegate is not specified.");
205  }
