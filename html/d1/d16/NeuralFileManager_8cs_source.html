764  public virtual void ServerCalculateApproximation()
765  {
766  if (!IsNeuralInputReady())
767  {
768  //Console.WriteLine();
769  //Console.WriteLine("WARNING: Neural approximation input data is not ready.");
770  //Console.WriteLine();
771  throw new InvalidOperationException("Neural input ready flag is not set.");
772  }
773  if (!IsNeuralBusy())
774  SetNeuralBusy();
775  if (IsNeuralOutputReady())
776  ClearNeuralOutputReady();
777  ClearNeuralInputReady();
778  IVector inputParameters = null;
779  ReadNeuralInput(ref inputParameters);
780  INeuralApproximator approximator = null;
781  LoadNeuralNetwork(ref approximator);
782  IVector outputValues = null;
783  // Calculate approximation:

785  WriteNeuralOutput(outputValues);
