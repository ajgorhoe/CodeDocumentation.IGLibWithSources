100  public Matrix3d(double[][] A)
101  {
102  if (A == null)
103  throw new ArgumentNullException("Jagged array to create a 3D matrix from is not specified (null reference).");
104  if (A.Rank != 2)
105  throw new ArgumentException("Jagged array to create a 3D matrix from is not a rank 2 array.");
106  if (A.Length != 3)
107  throw new ArgumentException("Jagged array to create a 3D matrix from does not have 3 rows.");
108  for (int i = 0; i < 3; ++i)
109  {
110  if (A[i].Length != 3)
111  throw new ArgumentException("Jagged array to create a 2D matrix from has inconsistent dimensions. Row "
112  + i + "has length " + A[i].Length + "Instead of 3.");
113  for (int j = 0; j < 3; ++j)
114  this[i, j] = A[i][j];
115  }
116  }

169  public static Matrix3d CreateFromColumns(IList<IVector> columnVectors)
170  {
171  if (columnVectors == null)
172  throw new ArgumentNullException("List of column vectors to create a 3D matrix from is not specified (null reference).");
173  if (columnVectors.Count != 3)
174  throw new ArgumentException("List of column vectors to create a 3D matrix from does not contain 3 vectors.");
175  int rows = columnVectors[0].Length;
176  int columns = columnVectors.Count;
177  if (rows!=3)
178  throw new ArgumentException("Length of column vectors to create a 3D matrix from is not 3.");
179  double[,] newData = new double[3,3];
180  for (int i = 0; i < rows; i++)
181  for (int j = 0; j < columns; j++)
182  newData[i,j] = columnVectors[j][i];
183  return new Matrix3d(newData);
184  }

187  public static Matrix3d CreateFromRows(IList<Vector> rowVectors)
188  {
189  if (rowVectors == null)
190  throw new ArgumentNullException("List of row vectors to create a 3D matrix from is not specified (null reference).");
191  int rows = rowVectors.Count;
192  int columns = rowVectors[0].Length;
193  if (rows != 3)
194  throw new ArgumentException("List of row vectors to create a 3D matrix from does not contain 3 vectors.");
195  if (columns != 3)
196  throw new ArgumentException("Length of row vectors to create a 3D matrix from is not 3.");
197  double[][] newData = new double[rows][];
198 
199  for (int i = 0; i < rows; i++)
200  {
201  newData[i] = rowVectors[i].ToArray();
202  }
203  return new Matrix3d(newData);
204  }
